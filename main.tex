\chapter{Modules}
Before we even start, MOST rings in this course are with unity and commutative! Basically what we are trying to do is study the structure of certain modules. Modules however, can be confusing and hard and horrible. So what we're gonna do is turn them into abelian groups, which is a lot easier, but we can't do that straight away so heres the plan: We replace a module by a projective resolution, apply $Hom(-, M)$ to obtain a cochain complex of abelian groups, and then take cohomology to measure the failure of exactness. Essentially, Homological algebra replaces modules by resolutions so that applying non-exact functors produces complexes whose cohomology records the obstruction to exactness. This wont make much sense now, but just keep this diagram in your head: 
\begin{align*}
    \text{Modules } M \xrightarrow{M \rightarrow(P_* \rightarrow M)} \text{Projective Resolutions } P_* \xrightarrow{Hom(P_*, M)} \\\text{Cochain complexes } \xrightarrow{H^n}  \text{Abelian Group } Ext^n(P_*, M)
\end{align*}
\section{Basic Definitions}
\begin{definition}
    Let $R$ be a ring. A \textbf{module} over $R$, or an $R$-module $M$, is an abelian group, usually written additively, together with an operation of $R$ on $M$, such that, for all $a, b \in R$, and $x, y \in M$ we have 
    \begin{align*}
        (a + b)x = ax + bx \text{ and } a(x + y) = ax + ay
    \end{align*}
    We also have that $a(-x) = -(ax)$ and $0x = 0$. 
\end{definition}
This is essentially a vector space where the scalars come from a ring rather than a field. We also have a similar idea of submodules. 

\begin{example}
    \begin{enumerate}
        \item If we take our ring to be a field, then modules over the field are simply vector spaces.
        \item If we take our set M to be a ring R. Then M = R is an R-module wherein the submodules are the ideals of R. 
        \item  Let R be a ring and n $\in \mathbb{Z}^n$, define 
        \begin{align*}
            R^n = \{(a_1, a_2, \dots, a_n) : a_i \in R\}
        \end{align*}
        We can make $R^n$ into an R-module by defining additionn and scalar multiplication componentwise. This module is called the \textbf{free module of rank n over R}.
        \item Let R = $\mathbb{Z}$ and let $A$ be any abelian group where the operation is written as $+$. We can make $A$ into a $\mathbb{Z}$-module as follows: for any $n \in \mathbb{Z}$ or $a \in A$ define
        \begin{align*}
            na = 
            \begin{cases}
            a + a + \cdots + a \quad (n \text{ times}), & \text{if } n > 0, \\
            0, & \text{if } n = 0, \\
            - a - a - \cdots - a \quad (-n \text{ times}), & \text{if } n < 0.
            \end{cases}                
        \end{align*}
        In this way, any abelian group is a $\mathbb{Z}$-module. 
    \end{enumerate}
\end{example}

\begin{definition}
    Let $M$ be an $R$-module. A subset $N \subseteq M$ is a \textbf{submodule} if:\begin{align*}
        0 \in N \\
        x, y \in N \implies x + y \in N \\
        r \in R \text{ and } x \in N \implies rx \in N
    \end{align*}
\end{definition}
Equivalently $N$ is a submodule iff $N$ is a subgroup of $(M,+)$ and is closed under scalar multiplication by the ring $R$.

\begin{definition}
    Let $M$ and $N$ be modules over $R$. Let $T : M \rightarrow N$ be a mapping, we say $T$ is \textbf{R-linear} when 
    \begin{align*}
        T(x + y)     &= T(x) + T(y) \, \forall x, y \in M\\
        T(\lambda x) &= \lambda T(x) \, \forall x \in M \, \forall \lambda \in R
    \end{align*}
    We denote the set of all $R$-linear maps from $M$ to $N$ as $Hom_R{M,N}$
\end{definition}

\begin{proposition}
    Hom$_R(M,N)$ is an $R$-module. 
\end{proposition}
\begin{proof}
    Define addition, scalar multiplication, and zero as follows: $(\alpha + \beta)(m) = \alpha(m) + \beta(m)$, $(\alpha \lambda)(m) = \alpha(m) \lambda$ for $\lambda \in R$, and $\mathbf{0}(m) = 0$.
    Then for $r \in R$ and $m_1, m_2 \in M$ we have 
    \begin{align*}
        (\alpha + \beta)(rm_1 + m_2) = \alpha(rm_1 + m_2) + \beta(rm_1 + m_2) = r\alpha(m_1) + \alpha(m_2) + r\beta(m_1) + \beta(m_2)
    \end{align*}
    which then after rearranging is equal to $r(\alpha + \beta)(m_1) + (\alpha + \beta)(m_2)$ since $\alpha$ and $\beta$ are already homs. Now let $m \in M$
    \begin{align*}
        (\alpha \lambda)(rm) = \alpha(rm)\lambda = (r \alpha(m))\lambda = r(\alpha(m)\lambda) = r((\alpha \lambda)(m))
    \end{align*}
    as required. Note that the associative property used in the third step required $R$ to be commutative\footnote{Let $m \in M$ and $a, b \in R$, $(ab)\alpha(m) = (\alpha(m))(ab)$ but $(a(b\alpha))(m) = (b\alpha)(m)a = (\alpha(m)b)a = \alpha(m)(ba)$ so $(ab) = (ba)$ for this to hold}.
\end{proof}

\begin{proposition}
    For any $\Lambda$-module $N$, $Hom_\Lambda(\Lambda, N) \cong N$
\end{proposition}
\begin{proof}
    \label{HomProp1}
    This is given by the isomorphism $\psi : Hom_\Lambda(\Lambda, N) \to N$ where $\psi(f) = f(1)$.
\end{proof}

\begin{proposition}
    Let $M,N$ be $R$-modules, and $T$ an $R$-linear map between $M$ and $N$ then 
    \begin{itemize}
        \item $Im(T)$ is a submodule of $N$ 
        \item $Ker(T)$ is a submodule of $M$ 
        \item $T$ is surjective iff $Im(T) = N$ 
        \item $T$ is injective iff $Ker(T) = M$ 
    \end{itemize}
\end{proposition}

\begin{definition}
    Let $M$ be a module over a ring $A$ and let $S$ be a subset of $M$. We say that $S$ is a \textbf{basis} of $M$ is $S$ is not empty, $S$ generates $M$, and $S$ is linearly independent. 
\end{definition}



\begin{definition}
    Let $M$ be a module over $R$, we call $M$ a \textbf{free} module when it has a basis.
\end{definition}

Obviously, every module has a generating set, because at worst you can take every single element of the module and put it in a set. So the generating set being linearly independent is the important distinguishing quality of free modules. 

In another sense, any ring is free over itself, since it can be generated by the unity element. 

The next definition is a property shared by all free modules, so although it reads like a proposition, it should be understood more as a definition. We'll start with something that should be familiar from rings. 

\begin{theorem}
    (Universal Property of Quotient Modules) Let $M$ be a $\lambda$-module, $N$ a submodule, and $\pi : M \to M/N$ by the projection map $\pi(m) = m + N$. Then if $f : M \to L$ is a $\Lambda$-linear map such that $N \subseteq ker(f)$, then there exists a unique $\hat{f} : M/N \to L$ such that $f = \hat{f} \circ \pi$
\end{theorem}

\begin{definition}\label{UnivPropFree}
    (Universal Property of free modules) Let $F$ be a free $R$-module generated by $X = \{e_1, e_2, \dots, e_n\}$ and $M$ any $R$-module with a map\footnote{Note: NOT necessarily a homomorphism} $f : X \to M$ then there exists a unique homomorphism $\hat{f} : F \to M$ thereby extending $f$. i.e.\footnote{Here we use for the first time whats known as a commutative diagram, which will be utilised extensively through this course. Remember to pay close attention to the direction of the arrows and note that a dashed arrow denotes the existence of a particular morphism given the statement of your proposition.}
    \[
    \begin{tikzcd}
    & F  \arrow[d, dashed, "\hat{f}"] & \\
    X \arrow[ru, "i"] \arrow[r, "f"] & M &
    \end{tikzcd}
    \]
    Where $i$ is the inclusion map. This is known as the universal property of free modules. 
\end{definition}

We'll now introduce the term 'resolution' which for now we can just picture as an unfolding of a module. 

\begin{theorem}
    \label{Free Theorem}
    Every $R$-module $M$ has a free resolution.
\end{theorem}
\begin{proof}
    Let $M$ be an $R$-module with generating set $\{m_i : i \in I\}$. Now let $F_0$ be the free module over $\{m_i: i \in I\}$. We have a natural surjective $R$-linear map $\epsilon: F_0 \to M$ (which we call the augmentation map). If $M$ itself is free, then having picked a basis as a generating set this is an isomorphism and we're done. If $M$ is not free this isn't an isomorphism and must have a non-empty kernel.\\ Let $K_0 = ker(\epsilon) \subseteq F_0$ with generating set $\{k_j: j \in J\}$ and $F_1$ be the free module over this generating set. Then, as before, there is a natural $R$-linear map $\partial_1 : F_1 \to F_0$ such that the image is $ker(\epsilon)$. Again, this may not be injective, so we might have to carry this process on infinitely so the proof is completed via induction. 
\end{proof}



\section{Short Sequences}
One of the most important tools for studying the structure of some algebraic object $B$ is the first isomorphism theorem, which relates whatever subobjects $B$ has to the possible homomorphic images of $B$. Written in the category of $R$-modules, we have:
\begin{theorem}
    (The First Isomorphism Theorem for Modules) Let $M,N$ be $R$-modules, and let $\varphi : M \rightarrow N$ be an $R$-module homomorphism. Then ker($\varphi$) is a submodule of $M$ and $M/ker(\varphi) \cong Im(\varphi)$
\end{theorem} 

When we've done this with rings before, we often start with a ring $B$ and then construct a homomorphism $\varphi$ (such that ker$(\varphi) \subseteq B$) to study the properties of $B$. Now, we want to consider given two modules $A, C$, whether there exists a module $B$ containing some isomorphic copy of $A$, such that $B/A \cong C$. In other words, $A$ is isomorphic to some submodule of $B$ if there is some injective homomorphism $\psi : A \rightarrow B$, as then $ A \cong Im(\psi) \subseteq B$. And $C$ is isomorphic to the resulting quotient $B/A$ if there is a surjective homomorphism $\varphi : B \rightarrow C$ with $ker(\varphi) = Im(\psi)$ as $B/ker(\varphi) \cong \varphi(B) = C$.

\begin{definition}
    Let $M_{n+1} \xrightarrow{T_{n+1}} M_n \xrightarrow{T_n} M_{n-1} \xrightarrow{T_{n-1}} \dots$
    be a sequence of $R$-modules joined together by $R$-linear maps. We say that the sequence is \textbf{exact} at $M_n$ when $Ker(T_n) = Im(T_{n+1})$
\end{definition}

\begin{proposition}
    Let $A$, $B$ and $C$ be $R$-modules over some ring $R$. Then 
    \begin{enumerate}
        \item The sequence $0 \rightarrow A \xrightarrow{\psi} B$ is exact at $A$ iff $\psi$ is injective.
        \item The sequence $B \xrightarrow{\varphi} C \rightarrow 0$ is exact at $C$ iff $\varphi$ is surjective.
    \end{enumerate}
\end{proposition}
\begin{proof}
    The (uniquely defined) homomorphism $0 \rightarrow A$ has image $0$ in $A$ due to the properties of homomorphisms. So this will be the kernel of $\psi$ iff $\psi$ is injective. Similarly, the kernel of the zero homomorphism $C \rightarrow 0$ is all of $C$, which is the image of $\varphi$ iff $\varphi$ is surjective. 
\end{proof}

\begin{definition}
    The exact sequence $0 \rightarrow A \xrightarrow{\psi} B \xrightarrow{\varphi} C \rightarrow 0$ is called a \textbf{short exact sequence}.
\end{definition}

Okay, so going back to our first isomorphism theorem, $M/ker(\varphi) \cong Im(\varphi)$ gives a short exacts sequence: 
\begin{align*}
    0 \rightarrow ker(\varphi) \xrightarrow{i} M \xrightarrow{\varphi} im(\varphi) \rightarrow 0
\end{align*}

\begin{example}
    \begin{enumerate}
        \item Given modules $A$ and $C$ we can always form their direct sum $B = A \oplus C$ and the sequence: 
        \begin{align*}
            0 \rightarrow A \xrightarrow{\varphi} A \oplus C \xrightarrow{\pi} C \rightarrow 0
        \end{align*}
        Where $\varphi(a) = (a,0)$ and $\pi(a,c) = c$. In particular it follows that there exists at least one extension of $C$ by $A$. 
        \item Consider the two $\mathbb{Z}$-modules $A = \mathbb{Z}$, $C = \mathbb{Z}/n\mathbb{Z}$:
        \begin{align*}
            0 \rightarrow \mathbb{Z} \xrightarrow{\varphi} \mathbb{Z} \oplus (\mathbb{Z}/n\mathbb{Z}) \xrightarrow{\pi} \mathbb{Z}/n\mathbb{Z} \rightarrow 0
        \end{align*}
        Is a special case of example 1. 
        \item Consider the same modules as above, we can get the exact sequence: 
        \begin{align*}
            0 \rightarrow \mathbb{Z} \xrightarrow{n} \mathbb{Z} \xrightarrow{\pi} \mathbb{Z}/n\mathbb{Z} \rightarrow 0
        \end{align*}
        wherein $n$ denotes the map $x \mapsto nx$ (obviously injective) and $\pi$ denotes the natural projection $ x \mapsto x (mod n)$ (obviously surjective).  
    \end{enumerate}
\end{example}
\begin{definition}
    Let $0 \rightarrow A \rightarrow B \rightarrow C  \rightarrow 0$ and $0 \rightarrow A' \rightarrow B' \rightarrow C'  \rightarrow 0$ be short exact sequences of modules. A \textbf{homomorphism of short exact sequences} is a triple $\alpha, \beta, \gamma$ of module homomorphisms such that the following diagram commutes: 
        \[
        \begin{tikzcd}
        0 \arrow[r] & A \arrow[r] \arrow[d, "\alpha"] 
        & B \arrow[r] \arrow[d, "\beta"] 
        & C \arrow[r] \arrow[d, "\gamma"] 
        & 0 \\
        0 \arrow[r] & A' \arrow[r] 
        & B' \arrow[r] 
        & C' \arrow[r] 
        & 0
        \end{tikzcd}
        \]
    Which is an \textbf{isomorphism of exact sequences} if $\alpha, \beta, \gamma$ are all isomorphisms. 
\end{definition}

Note that in such a scenario as above, if $\alpha$, $\beta$ and $\gamma$ are isomorphisms and the diagram commutes then the top sequence is exact iff the bottom sequence is exact. 

\begin{proposition}
    (The Short Five lemma) Let $\alpha, \beta, \gamma$ be a homomorphism of exact sequences
        \[
        \begin{tikzcd}
        0 \arrow[r] & A \arrow[r] \arrow[d, "\alpha"] 
        & B \arrow[r] \arrow[d, "\beta"] 
        & C \arrow[r] \arrow[d, "\gamma"] 
        & 0 \\
        0 \arrow[r] & A' \arrow[r] 
        & B' \arrow[r] 
        & C' \arrow[r] 
        & 0
        \end{tikzcd}
        \]
    \begin{enumerate}
        \item if $\alpha$ and $\gamma$ are injective, then so is $\beta$
        \item if $\alpha$ and $\gamma$ are surjective, then so is $\beta$
        \item if $\alpha$ and $\gamma$ are isomorphisms, then so is $\beta$, and thus the two sequences are isomorphic. 
    \end{enumerate}   
\end{proposition}

\begin{definition}
    Let $\mathcal{E} = 0 \rightarrow A \xrightarrow{j} B \xrightarrow{p} C \rightarrow 0$ be an exact sequence. We say that $\mathcal{E}$ \textbf{splits} if there exists a commutative diagram: 
        \[
        \begin{tikzcd}
        0 \arrow[r] & A \arrow[r, "j"] \arrow[d, "id_A"] 
        & B \arrow[r, "p"] \arrow[d, "\varphi"] 
        & C \arrow[r] \arrow[d, "id_C"] 
        & 0 \\
        0 \arrow[r] & A \arrow[r, "\iota"] 
        & A \oplus C \arrow[r, "\pi"] 
        & C \arrow[r] 
        & 0
        \end{tikzcd}
        \]
    i.e. there exists such a $\varphi : B \rightarrow A \oplus C$ such that $\varphi \circ j = \iota$ and $\pi \circ \varphi = p$. It follows that $\varphi$ is an isomorphism from the short five lemma. 
\end{definition}

\begin{definition}
    Let $\mathcal{E}$ be a short exact sequence as before. We say that $\mathcal{E}$ \textbf{splits on the right} if there exists some $\lambda$-homomorphism $s : C \rightarrow B$ such that $p \circ s = id_C$. I.e. we have the following diagram: 
    \[
    \begin{tikzcd}
    0 \arrow[r]
    & A \arrow[r, "j"]
    & B \arrow[r, "p"]
    & C \arrow[r]
    & 0
    \arrow[from=1-4, to=1-3, dotted, "s"', shift left=3]
    \end{tikzcd}
    \]
    We say $\mathcal{E}$ \textbf{splits on the left} when $\exists r : B \rightarrow A$ such that $r \circ j = id_A$.
\end{definition}

\begin{lemma}
    Let $\mathcal{E} = 0 \rightarrow A \xrightarrow{j} B \xrightarrow{p} C \rightarrow 0$ be an exact sequence of $\lambda$-modules. Then $\mathcal{E}$ splits on the right $\iff$ $\mathcal{E}$ splits $\iff \mathcal{E}$ splits on the left. 
\end{lemma}
\begin{proof}
    (Splits on right $\implies$ splits )
    Since $\mathcal{E}$ splits on the right, $\exists s : C \rightarrow B$ such that $p \circ s = id_C$. Consider the two exact sequences:
        \[
        \begin{tikzcd}
        0 \arrow[r] & A \arrow[r, "\iota"] \arrow[d, "id_A"] 
        & A \oplus C \arrow[r, "\pi"] \arrow[d, "\psi"]
        & C \arrow[r] \arrow[d, "id_C"]
        & 0 \\
        0 \arrow[r] & A \arrow[r, "j"]  
        & B \arrow[r, "p"] 
        & C \arrow[r] 
        & 0 
        \end{tikzcd}
        \]
    where $\psi(a,c) = j(a) + s(c)$. Note that we can always construct the exact sequence at the top as shown in examples. Observe: 
    \begin{align*}
        p \psi(a,c) &= p(j(a) + s(c))\\
                    &= p(j(a)) + p(s(c))
    \end{align*}
    Since the sequence is exact, $p(j(a)) = 0$ so 
    \begin{align*}
        p \psi(a,c) &= 0 + p(s(c))\\
                    &= c \\
                    &= \pi(a,c)
    \end{align*}
    So the diagram: 
        \[
        \begin{tikzcd}
        A \oplus C \arrow[r, "\pi"] \arrow[d, "\psi"'] 
        & C \arrow[d, "id_C"] \\
        B \arrow[r, "p"'] 
        & C
        \end{tikzcd}
        \]
    commutes. Similarly, the other side of the original diagram commutes. So $\mathcal{E}$ splits.\\

    \noindent (Splits implies splits on the right)
    Assume that we have the following commutative diagram: 
        \[
        \begin{tikzcd}
        0 \arrow[r] & A \arrow[r, "j"] \arrow[d, "id_A"] 
        & B \arrow[r, "p"] \arrow[d, "\varphi"] 
        & C \arrow[r] \arrow[d, "id_C"] 
        & 0 \\
        0 \arrow[r] & A \arrow[r, "\iota"] 
        & A \oplus C \arrow[r, "\pi"] 
        & C \arrow[r] 
        & 0
        \end{tikzcd}
        \]
        By the five lemma, $\varphi$ is an isomorphism, so there is an inverse $\varphi^{-1} : A \oplus C \to B$. Let $f : C \to A \oplus C$ be the inclusion map. Define $s: C \to B$ where $ s = \varphi^{-1} \circ f$. Then we have 
        \begin{align*}
            p \circ s = p \circ \varphi^{-1} \circ f = \pi \circ f = id_C
        \end{align*}
        Since $\pi \circ \varphi = p$ as the diagram commutes.\\
        \noindent (Splits on the left implies splits) Assume there exists some $r : B \to A$ such that $r \circ j = id_A$. Then consider the exact two exact sequences: 
        \[
        \begin{tikzcd}
        0 \arrow[r] & A \arrow[r, "j"] \arrow[d, "id_A"] 
        & B \arrow[r, "p"] \arrow[d, "\varphi"] 
        & C \arrow[r] \arrow[d, "id_C"] 
        & 0 \\
        0 \arrow[r] & A \arrow[r, "\iota"] 
        & A \oplus C \arrow[r, "\pi"] 
        & C \arrow[r] 
        & 0
        \end{tikzcd}
        \]
        Then define $\varphi : B \to A \oplus C$ by $\varphi(b) = (r(b), p(b))$. Observe: 
        \begin{align*}
            \varphi \circ j (a) &= (r(j(a)), p(j(a)))\\
                                &= (a, 0)\\
                                &= \iota(a)
        \end{align*}
        Since the sequence is exact. Thus the diagram commutes and $\varphi$ is an isom by the five lemma. The final implication is very similar to the other case. 
\end{proof}

\subsection{Projective Modules}
Here we will lay out 3 seperate definitions of projective modules, and then show they are equivalent. There are actually more ways to characterise projective modules, which we will learn about later. 

\begin{definition}(1. Lifting/Universal Property)\label{ProjLift}
    Suppose we are given a diagram of $\lambda$-modules: 
    \[
    \begin{tikzcd}
    & P \arrow[ld, dashed, "\hat{\varphi}"'] \arrow[d, "\varphi"] & \\
    M \arrow[r, two heads, "p"] & N &
    \end{tikzcd}
    \]
    P is \textbf{projective} when $\exists \hat{\varphi} : P \rightarrow M$ such that $p \circ \hat{\varphi} = \varphi$. I.e. if for every surjective map $p : M \to N$ and $R$-linear map $\varphi : P \to N$ there exists a lift $\hat{\varphi} : P \to M$ such that $p \circ \hat{\varphi} = \varphi$. 
\end{definition}

Despite the fact we've referred to this definition as the \textit{universal property}, it is not in-fact universal, since $\hat{\varphi}$ does not have to be unique! We just call it that because it resembles the universal property of free modules. 

\begin{definition}(2. Splitting Property)
    Suppose we are given an exact sequence of $\lambda$-modules 
    \begin{align*}
        0 \rightarrow K \xrightarrow{j} X \xrightarrow{p} P \rightarrow 0
    \end{align*}
    Then $P$ is \textbf{projective} iff every such exact sequence splits. 
\end{definition}

\begin{definition}(3. Summand Property)
    $P$ is \textbf{projective} $\iff$ there exists a $\lambda$-module $K$ such that $K \oplus P$ is free (i.e. has a basis).
\end{definition}

\begin{proposition}
    Definition 1 $\iff$ Definition 2 $\iff$ Definition 3
\end{proposition}
\begin{proof}
    $(1) \implies (2)$
    Given an exact sequence: 
    \begin{align*}
        0 \rightarrow K \xrightarrow{j} X \xrightarrow{p} P \rightarrow 0
    \end{align*}
    We can get the diagram: 
    \[
    \begin{tikzcd}
    & P \arrow[ld, dashed, "s \, =\,  \hat{id}"'] \arrow[d, "id"] & \\
    X \arrow[r, two heads, "p"] & P &
    \end{tikzcd}
    \]
    As $p$ is surjective by exactness and by Definition 1, $\exists s : P \rightarrow X$ (s.t. $s \, = \, \hat{id}$) wherein $p \circ s = id_p$. This is the definition of splitting on the right, which implies that the exact sequence splits.\\
    $(2) \implies (3)$ Let $P$ be a $\lambda$-module and let $(e_i)_{i \in I}$ be a set of generators for $P$. i.e. $\forall x \in P$ there exists $(x_i)_{i \in I} \in \lambda$ (almost all zero), such that $x = \sum e_i x_i$. Now let $(E_i)_{i \in I}$ be a basis for the free module $F$. The mapping $p : F \to P$ given by $p(E_i) = e_i$ is clearly surjective, so setting $K = ker(p)$ we have the exact sequence
    \begin{align*}
        0 \rightarrow K \xrightarrow{i} F \xrightarrow{p} P \rightarrow 0
    \end{align*} 
    Since we are assuming definition 2, this exact sequence must split, so $F \cong K \oplus P$ where $F$ is free. \\
    $(3) \implies (1)$ Given a diagram: 
    \[
    \begin{tikzcd}
    & P \arrow[ld, dashed, "?"'] \arrow[d, "\varphi"] & \\
    M \arrow[r, two heads, "p"] & N &
    \end{tikzcd}
    \]
    We need to produce a $\hat{\varphi} : P \to M$. Since we are assuming definition 3, we have that $\exists K$ such that $ P \oplus K$ is free. Let $(E_i)_{i \in I}$ be the basis of this module and modify $\varphi : P \to N$ as follows: \\
    Define $\psi : P \oplus K \to N$ by $\psi(x,y) = \varphi(x)$ where $x \in P$ and $y \in K$. Since $p$ is surjective, for each $i \in I$ choose $\xi_i$ such that $p(\xi_i) = \psi(E_i) \in N$.\\
    Now define $\hat{\psi(E_i)} = \xi_i$. What we've then done is: 
    \[
    \begin{tikzcd}
    & P \oplus K  \arrow[ld, dashed, "\hat{\psi}"'] \arrow[d, "\psi"] & \\
    M \arrow[r, two heads, "p"] & N &
    \end{tikzcd}
    \] 
    Which commutes simply by the universal property of free modules.\\
    Then define $\hat{\varphi} : P \to M$ by $\hat{\varphi}(x) = \hat{\psi}(x,0)$. Then
    \begin{align*}
        p \hat{\phi} (x) = p \hat{\psi} (x,0) = \psi(x,0) = \varphi(x)
    \end{align*}
    So the diagram commutes. 
\end{proof}

\begin{proposition}
    Let $M$ be a free module, then $M$ is projective. 
\end{proposition}
\begin{proof}
    According to \ref{ProjLift} given a free module $F_X$, $R$-linear map $\phi: F_X \to N$ and surjective $R$-linear map $\alpha : M \to N$ we need to produce a $\hat{\phi} : F_X \to M$ s.t. $\alpha \circ \hat{\phi} = \phi$.\\ 
    Let $X = \{e_i : i \in I \}$ be a basis of $F_X$. Observe that $\{\phi(e_i) : i \in I\} \subset N$. Since $\alpha$ is surjective, let ${m_i : i \in I} \subset M$ satisfy $\alpha(m_i) = \phi(e_i)$. Now define $\hat{\phi}(e_i) = m_i$. Thus we have the following diagram
    \[
    \begin{tikzcd}
    & F_X \arrow[ld, dashed, "\hat{\phi}"'] \arrow[d, "\phi"] & \\
    M \arrow[r, two heads, "\alpha"] & N &
    \end{tikzcd}
    \] 
    This diagram commutes by construction, so we just need to prove that $\hat{\phi}$ is a valid $R$-linear map.Define a set map $f : X \to M$ given by $f(e_i) = m_i$ as $\alpha$ is surjective, hence we have the following diagram: 
    \[
    \begin{tikzcd}
        & F_X \arrow[d, dashed, "\hat{\phi}"] \\
        X \arrow[ur, "i"] \arrow[r, "f"] & M
    \end{tikzcd}
    \]
    Where $i$ is the inclusion map. By the universal property of free modules \ref{UnivPropFree}, $\hat{\phi}: F_X \to M$ is an $R$-linear map. 
\end{proof}

\begin{proposition}
    Let $M$ be a free module, then $M$ is projective. 
\end{proposition}
\begin{proof}
    This is normally proved via the lifting property as above, but you can shortcut this by using the equivalent summand definition of a projective module. For any module we have $M \cong M \oplus 0$, so therefore if $M$ is free it is also the summand of a free module and so is projective. 
\end{proof}

\subsection{The Hom Functor}
This section gives some broad details in the Hom functor. Let $R$ be a ring and suppose that the $\Lambda$-module $M$ is an extension of $N$ by $L$, given us the corresponding exact sequence of modules: 
\begin{align*}
    0 \rightarrow L \xrightarrow{\psi} M \xrightarrow{\phi} N \rightarrow 0 
\end{align*}
We could want to ask whether given a $\Lambda$-module homomorphism from some fixed $\Lambda$-module $D$ to either $L$ or $N$ implies that there is an induced homomorphism from $D$ to $M$. As it turns out, if we have a homomorphism from $D$ to $L$ we can get one from $D$ to $M$ immediately. 

\begin{proposition}
    Let $0 \rightarrow L \xrightarrow{\psi} M \xrightarrow{\phi} N \rightarrow 0$ be a short exact sequence of $\Lambda$-modules. If $f \in Hom_\Lambda(D,L)$, then the composite $f* = \psi \circ f$ is an $\Lambda$-module homomorphism from $D$ to $M$. 
\end{proposition}

\begin{proof}
    Observe the diagram: 
    \[
    \begin{tikzcd}
    D \arrow[d, "f"'] \arrow[rd, "f'", dashed] &   \\
    L \arrow[r, "\psi"]                        & M
\end{tikzcd}
    \]
    The composition of $\Lambda$-homs are $\Lambda$-homs. 
\end{proof}

Putting this another way, composition with $\psi$ induces the map $\psi^* : Hom_\Lambda(D,L) \to Hom_\Lambda(D,M)$ given by $f \mapsto f^* = \psi \circ f$.\\

More importantly, it preserves some degree of exactness: 


\begin{proposition}
    Let $D$, $L$, and $M$ be $\Lambda$-modules, and let $\psi : L \to M$ be a $\Lambda$-module homomorphism. Then the induced map $\psi^*$ as described above is a homomorphism of abelian groups. Moreover, if $\psi$ is injective, then $\psi^*$ is injective, i.e. if 
    \begin{align*}
        0 \rightarrow L \xrightarrow{\psi} M \text{ is exact}\\
        0 \rightarrow Hom_\Lambda(D,L) \xrightarrow{\psi^*} Hom_\Lambda(D,M) \text{ is exact}
    \end{align*}
    Thus in category theoretic terms, this is to say that $Hom_\Lambda(D,-)$ is left exact. 
\end{proposition}
\begin{proof}
    The fact its a hom is immediate. If $\psi$ is injective then given distinct $f$ and $g \in Hom_\Lambda(D,L)$ we are going to get distinct homs $\psi \circ f $ and $\psi \circ g$. 
\end{proof}

Easy enough. Now finding a homomorphism from $D$ to $M$ given one from $D to N$ is a lot harder! As before, we essentially want to find some $F$ such that the following diagram commutes: 

\[
\begin{tikzcd}
& D \arrow[d, "f"] \arrow[dl, "F"', dashed] \\
M \arrow[r, "\varphi"'] & N
\end{tikzcd}
\]

Or put another way, like before, composition with $\phi$ induces a homomorphism of abelian groups. $\phi^* : Hom_\Lambda(D,M) \to Hom_\Lambda(D,N)$ given by $F \mapsto F' = \phi \circ F$. In terms of $\phi^*$, the hom $f$ into $N$ lifts to a homomorphism to $M$ iff $f$ is in the image of $\phi^*$. This may not always be possible!!

\begin{proposition}
    Given an exact sequence $M \xrightarrow{\phi} N \rightarrow 0$, $ Hom_\Lambda(D,M) \xrightarrow{\phi^*} Hom_\Lambda(D,N) \rightarrow 0$ is not necessarily exact.  
\end{proposition}
\begin{proof}
    construct counterexample with nonsplit sequence. 
\end{proof}

However, all is not lost! We have the following theorem: 

\begin{theorem}
    If $0 \rightarrow L \xrightarrow{\psi} M \xrightarrow{\phi} N \rightarrow 0$ is a short exact sequence of $\Lambda$-modules, then $0 \rightarrow Hom_\Lambda(D,L) \xrightarrow{\psi^*} Hom_\Lambda(D,M) \xrightarrow{\phi^*} Hom_\Lambda(D,N)$ is exact. 
\end{theorem}
\begin{proof}
    From a previous proposition we have that  $0 \rightarrow Hom_\Lambda(D,L) \xrightarrow{\psi^*} Hom_\Lambda(D,M)$ is exact, so it remains to show that Im($\psi_*$) = Ker($\phi_*$). 
    First Im($\psi_*$) $\subseteq$ Ker($\phi_*$) since $\phi_*(\psi_*(g)) = \phi \circ \psi \circ g = 0$ by exactness. Now observe 
    \begin{align*}
        Ker(\phi_*) &= \{f \in Hom_\Lambda(D,M): \phi_*(f) = 0\}\\
                    &= \{f \in Hom_\Lambda(D,M): \phi \circ f = 0\}
    \end{align*}
    So Im($f$) $\subseteq$ ker($\phi$). By exactness of the original sequence we thus have Im($f$) $\subseteq$ Im($\psi$). Consider $f(d) \in M$, $f(d) \in$ Im($\psi$) so since $\psi$ is injective there exists exactly one $l \in L$ such that $\psi(l) = f(d)$. Define a new map $h(d) = l$. Then $\psi_*(h)(d) = \psi \circ h (d) = \psi(l) = f(d)$. So $f \in$ Im($\psi_*$). Hence Ker($\phi_*$) $\subseteq$ Im($\psi_*$).
    Thus we have equality and the sequence is exact.
\end{proof}

We can phrase this in category theoretic terms\footnote{see appendix for definitions}. 

\begin{proposition}
    The functor Hom$_\Lambda(D, -)$ is left exact. 
\end{proposition}

Note that the above proposition also applies to the contravariant functor $Hom_\Lambda(-, D)$.

\begin{theorem}
    If $0 \rightarrow L \xrightarrow{\psi} M \xrightarrow{\phi} N \rightarrow 0$ is an exact sequence, and $D$ is a projective module, then $0 \rightarrow Hom_\Lambda(D,L) \xrightarrow{\psi^*} Hom_\Lambda(D,M) \xrightarrow{\phi^*} Hom_\Lambda(D,N) \rightarrow 0$ is exact!
\end{theorem}
\begin{proof}
    By previous theorem we have that $0 \rightarrow Hom_\Lambda(D,L) \xrightarrow{\psi^*} Hom_\Lambda(D,M) \xrightarrow{\phi^*} Hom_\Lambda(D,N)$ is exact, so it remains to show that $\phi^*$ is surjective. Let $f \in Hom_\Lambda(D,N)$, we know that $\phi$ is a surjection and that $D$ is a projective module, so we have the following diagram
    \[
    \begin{tikzcd}
    & D \arrow[ld, dashed, "\hat{f}"'] \arrow[d, "f"] & \\
    M \arrow[r, two heads, "\phi"] & N &
    \end{tikzcd}
    \]
    I.e. $\exists \hat{f}: D \to M$ such that $\phi \circ \hat{f} = f$. Observe now that $\phi^*(\hat{f}) = \phi \circ \hat{f} = f$. Hence $\phi^*$ is surjective. 
\end{proof}

We can phrase this in category theoretic terms, giving yet ANOTHER "definition" of projective modules. 
\begin{corollary}
    Let $D$ be a $\Lambda$-module. Then $Hom_\Lambda(D, -)$ is exact iff $D$ is projective. 
\end{corollary}

The above theorem does not apply to the contravariant Hom functor. We need $D$ to be an injective module for $Hom_\Lambda(-,D)$ to be exact, although we will not go into the definition of an injective module. 

We will see later, the study of how much $Hom_\Lambda(D, -)$ fails to be exact is one of the big areas of homological algebra. Note we have essentially the same results for the contravariant functor $Hom_\Lambda(-, D)$ in terms of failure to be exact and so on. We will end this section with a useful proposition about $Hom$.

\begin{proposition}
    Let $\Lambda$ be a ring and $A$ be an arbitrary $\Lambda$-module, then 
    \begin{align*}
        Hom_\Lambda(\Lambda, A) \cong A
    \end{align*}
\end{proposition}
\begin{proof}
    Define a $\Lambda$ homomorphism $\varphi : Hom_\Lambda(\Lambda, A) \to A$ which evaluates $f \in Hom_\Lambda(\Lambda, A)$ at $1 \in A$. Immediately well-defined and an isomorphism from definitions. 
\end{proof}



\subsection{Resolutions}

\begin{definition}
    Let $M$ be any $R$-module. A \textbf{projective resolution} of A is an exact sequence 
    \begin{align*}
        \dots \rightarrow P \xrightarrow{d_n} P_{n-1} \xrightarrow{d_{n-1}} \dots \xrightarrow{d_1} P_0 \xrightarrow{\epsilon} M \rightarrow 0.  
    \end{align*}
    such that each $P_i$ is a projective $R$-module.  
\end{definition}

\begin{proposition}
    Every $R$-module $M$ has a projective resolution. 
\end{proposition}
\begin{proof}
    Let $P_0$ be any free (hence projective) $R$-module on a set of generators of $M$ and define an $R$-module hom $\epsilon$ from $P_0$ onto $M$ as in Theorem \ref{Free Theorem}. This begins the resolution $\epsilon : P_0 \xrightarrow{\epsilon} M \rightarrow 0$. Since $\epsilon$ is surjective, the sequence is exact. Next let $K_0 = ker(\epsilon)$ and let $P_1$ be any free module mapping itself onto the submodule $K_0 \subseteq P_0$, thus giving the second stage $P_1 \rightarrow P_0 \rightarrow M \rightarrow 0$ which is also exact by construction. Carrying this process on to the $n$th stage, gives a free resolution of $M$, and thus a projective resolution.    
\end{proof}

There are many reasons we use projective resolutions, but one important reason is so that we can lift $R$-maps of modules to chain mappings.

\begin{definition}
    Let $M$ and $M'$ be $\Lambda$-modules and $P_* \rightarrow M$ and $Q_* \rightarrow M'$ be projective resolutions of $M$ and $M'$ respectively. Given an $R$-linear map $f : M \to M'$, a lifting is a sequence of $R$-linear maps $f_* = (f_r)_{0 \leq r}$ such that $f_r : P_r \to Q_r$ wherein the diagram 
    \[
    \begin{tikzcd}
    \dots \arrow[r] & P_n \arrow[r, "\partial_n"] \arrow[d, "f_n"] 
    & P_{n-1} \arrow[r, "\partial_{n-1}"] \arrow[d, "f_{n-1}"] 
    & \dots \arrow[r]
    & P_0 \arrow[r, "\epsilon"] \arrow[d, "f_0"]
    & M \arrow[r] \arrow[d, "f"]
    & 0 \\
    \dots \arrow[r] & Q_n \arrow[r, "\partial_n'"] 
    & Q_{n-1} \arrow[r, "\partial_{n-1}'"]
    & \dots \arrow[r]
    & Q_0 \arrow[r, "\epsilon'"]
    & M' \arrow[r] 
    & 0
    \end{tikzcd}
    \]
    commutes
\end{definition}

\begin{proposition}\label{Lifting of Chain Complexes}
    Define $M, M', P_*, Q_*, \text{ and } f$ as above. Then there always exists a lifting of $f$ to $f_*$
\end{proposition}
\begin{proof}
    Start by constructing $f_0 : P_0 \to Q_0$. We have 
    \[
    \begin{tikzcd}
        P_0 \arrow[r, "\epsilon"]  & M \arrow[r] \arrow[d, "f"]
        & 0\\
        Q_0 \arrow[r, "\epsilon'"] & M' \arrow[r] 
        & 0 
    \end{tikzcd}
    \]
    Since $P_0$ is projective, $\exists f_0 : P_0 \to Q_0$ as in 
    \[
    \begin{tikzcd}
    & P_0 \arrow[ld, dashed, "f_0"] \arrow[d, "f"] & \\
    Q_0 \arrow[r, two heads, "\epsilon'"] & M' &
    \end{tikzcd}
    \]
    Making the original diagram commute. Suppose then that we can lift $f$ to $f_{n-1}$ giving us: 
    \[
    \begin{tikzcd}
    \dots \arrow[r] & P_n \arrow[r, "\partial_n"]  
    & P_{n-1} \arrow[r, "\partial_{n-1}"] \arrow[d, "f_{n-1}"]
    & P_{n-2} \arrow[r] \arrow[d, "f_{n-2}"] 
    & \dots \arrow[r]
    & P_0 \arrow[r, "\epsilon"] \arrow[d, "f_0"]
    & M \arrow[r] \arrow[d, "f"]
    & 0 \\
    \dots \arrow[r] & Q_n \arrow[r, "\partial_n'"] 
    & Q_{n-1} \arrow[r, "\partial_{n-1}'"]
    & Q_{n-2} \arrow[r]
    & \dots \arrow[r]
    & Q_0 \arrow[r, "\epsilon'"]
    & M' \arrow[r] 
    & 0
    \end{tikzcd}
    \]
    Note that $\partial_{n-1}' f_{n-1} \partial_n = f_{n-2}\partial_{n-1}\partial{n}$ by diagram chasing. Since $\partial_{n-1}\partial_n = 0$ by exactness, we have that the LHS of this equation is 0, hence $Im(f_{n-1}\partial_n) \subset Ker(\partial_{n-1}')$. Moreover, by exactness, $Ker(\partial_{n-1}') = Im(\partial_n')$ Hence we have the following diagram: 
    \[
    \begin{tikzcd}
    & P_n \arrow[ld, dashed, "f_n"] \arrow[d, "f_{n-1}\partial_n"] & \\
    Q_n \arrow[r, two heads, "\partial_n'"] & Im(\partial_n') &
    \end{tikzcd}
    \]
    Which then implies that $\exists f_n: P_n \to Q_n$ such that the diagram commutes $(\partial_n' \circ f_n = f_{n-1}\circ \partial_n)$ as $P_n$ is projective. 
\end{proof}

\section{Natural Long Exact Sequences}
\subsection{The Coefficient Exact Sequence}
In this section, we will be showing that if we start with a short exact sequence of $\Lambda$-modules 
\begin{align*}
    0 \rightarrow A \xrightarrow{f} B \xrightarrow{g} C \rightarrow 0
\end{align*}
    and a projective resolution $P_*$ for some $\Lambda$-module $M$, by applying the covariant functor $Hom_\Lambda(P_*, -)$ to each element of the exact sequence, we can induce a long exact sequence of cohomology groups, called the \textbf{coefficient exact sequence}
\begin{definition}
    Let $C^*$ be a sequence of $\Lambda$-module homomorphisms: 
    \begin{align*}
        C^* = (0 \rightarrow C^0 \xrightarrow{\delta^0} C^1 \xrightarrow{\delta^1} \dots \rightarrow C^{n-1} \xrightarrow{\delta^{n-1}} C^n \xrightarrow{\delta^{n}} \dots)
    \end{align*}    
    \begin{itemize}
        \item $C^*$ is a cochain complex if for any $n$, $\delta^{n+1} \circ \delta^{n} = 0$. 
        \item If $C^*$ is a cochain complex, its nth \textbf{cohomology group} is the quotient group $Ker (\delta^{n})/Im(\delta^{n-1})$, and is denoted by $H^n(C^*)$.  
    \end{itemize}
\end{definition}

By contrast, chain complexes are defined where the index decreases between modules:

\begin{definition}
    Let $C_*$ be a sequence of $\Lambda$-module homomorphisms:
    \begin{align*}
        C_* = (\dots \xrightarrow{\partial_{n+1}} C_n \xrightarrow{\partial_n} C_{n-1} \rightarrow \dots \rightarrow C_1 \xrightarrow{\partial_1} C_0 \rightarrow 0)
    \end{align*}
    \begin{itemize}
        \item $C_*$ is a chain complex if for any $n$, $\partial_n \circ \partial_{n+1} = 0$
        \item If $C_*$ is a chain complex, then its $n$th \textbf{homology group} is the quotient group $H_n(C_*) = Ker(\partial_n) / Im(\partial_{n+1})$
    \end{itemize}
\end{definition}

\begin{remark}
    Note that if $C^*$ is a cochain (or chain) complex, then $C^*$ is an exact sequence iff all its cohomology groups are zero. In this way the nth cohomology group measures the failure of exactness at the nth stage! Moreover note that we will use superscript for cochain complexes, and subscript for chain complexes. 
\end{remark}

\begin{proposition}
    Let $M$ be a $\Lambda$-module and $P_*$ be a projective resolution of $M$ such that
    \begin{align*}
        \dots \rightarrow P_n \xrightarrow{\partial_n} P_{n-1} \xrightarrow{\partial_{n-1}} \dots \xrightarrow{\partial_1} P_0 \xrightarrow{\epsilon} M \rightarrow 0
    \end{align*}
    Then this sequence is a chain complex.
\end{proposition}
\begin{proof}
    Immediate since a projective resolution is exact so $Ker(\partial_n) = Im(\partial_{n+1})$ hence $\partial_n \circ \partial_{n+1} = 0$ 
\end{proof}
In this sense, a lifting between two resolutions is a chain mapping.
\begin{definition}
    Let $A^*$, $B^*$ and $C^*$ all be cochain complexes. A \textbf{short exact sequence} of complexes $0 \rightarrow A^* \xrightarrow{\alpha} B^* \xrightarrow{\beta} C^* \rightarrow 0$ is a sequence of homomorphisms of complexes such that $0 \rightarrow A^n \xrightarrow{\alpha^n} B^n \xrightarrow{\beta^n} C^n \rightarrow 0$ is short exact for every n. 
\end{definition}

The most important thing, for us, about cochain complexes is that they give rise to long exact sequences in cohomology. 

\begin{theorem}(Algebraic Mayer-Vietoris Theorem, Cochain Version)
    Let $0 \rightarrow A^* \xrightarrow{\alpha} B^* \xrightarrow{\beta} C^* \rightarrow 0$  be a short exact sequence of cochain complexes. Then there is a long exact sequence of cohomology groups: 
    \begin{align*}
        0 \rightarrow H^0(A^*) \rightarrow H^0(B^*) \rightarrow H^0(C^*) \xrightarrow{\delta^0} H^1(A^*)\\
        \rightarrow H^1(B^*) \rightarrow H^1(C^*) \xrightarrow{\delta^1} H^2(A^*) \rightarrow \dots 
    \end{align*}
    Where the maps between cohomology groups are those induced by the induced by the homomorphisms of cochain complexes. Note that the maps $\delta^n$ are called \textbf{connecting homomorphisms}.
\end{theorem}

\begin{proof}
    The proof for this is somewhat lengthy and beyond the examinable purview of this course so I will (along with some other theorems I will go onto state) cast it out to the appendix. Note that for each $n$ verifying that  $H^n(A^*) \rightarrow H^n(B^*) \rightarrow H^n(C^*)$ is exact is rather straightforward; the challenge lies in constructing the connecting homomorphisms, showing that they are in fact homomorphisms, and moreover that they are exact.  

\end{proof}


\begin{proposition}
    Let $P_*$ be the projective resolution of the $R$-module $M$ given by 
    \begin{align*}
        P_* = (\dots \rightarrow P \xrightarrow{d_n} P_{n-1} \xrightarrow{d_{n-1}} \dots \xrightarrow{d_1} P_0 \xrightarrow{\epsilon} M \rightarrow 0.)    
    \end{align*}
    and let $N$ be some coefficient $R$-module. Then we can induce the cochain complex
    \begin{align*}
        Hom_R(P_*, N) = (0 \rightarrow Hom_R(M,N) \xrightarrow{\epsilon^*} Hom_R(P_0, N) \xrightarrow{d_1^*} Hom_R(P_1, N) \xrightarrow{d_2^*} \dots \\
        \dots \xrightarrow{d_{n-1}^*} Hom_R(P_{n-1}, N) \xrightarrow{d_n^*} Hom_R(P_n, N) \rightarrow \dots)
    \end{align*}
\end{proposition}

\begin{proof}
    Simply just take the left exact contravariant functor $Hom_R(-,N)$. The left exactness implies that $d^*_{n+1} d_n^* = 0$ for all $n$. Note that this sequence is not necessarily exact. 
\end{proof}



\noindent Okay, we are almost there!

\begin{theorem}
    Let $P_*$ be a free (thus projective) resolution of the $R$-module $M$. Moreover let $0 \rightarrow A \xrightarrow{\iota} B \xrightarrow{p} C \rightarrow 0$. 
    be an exact sequence of $R$-modules. Then we have 
    \begin{align*}
        0 \rightarrow Hom(P_*, A) \rightarrow Hom(P_*, B) \rightarrow Hom(P_*, C) \rightarrow 0
    \end{align*}
    is a short exact sequence of cochain complexes. 
\end{theorem}
\begin{proof}
    $Hom_\Lambda(P_*, -)$ is a covariant functor, so we get
    \begin{align*}
        0 \rightarrow Hom(P_*, A) \rightarrow Hom(P_*, B) \rightarrow Hom(P_*, C) \rightarrow 0
    \end{align*}
    As a sequence. Since $P_i$ is projective for all $i$, this sequence is exact. 
\end{proof}

\noindent All this culminates in what we originally set out to do

\begin{remark}
    Notation: Where $M$ is an $R$-module and $P_*$ is a free resolution of $M$, denote 
    \begin{align*}
        H^n(M, N) := H^n(Hom(P_*, N))
    \end{align*}
    i.e. the cohomology groups $H^n(M,N)$ is the cohomology of the hom complex. 
\end{remark}

\begin{corollary}(Coefficient Exact Sequence)
    Take a $\Lambda$-module $M$ and let $0 \rightarrow A \xrightarrow{\iota} B \xrightarrow{p} C \rightarrow 0$ be an exact sequence of $\Lambda$-modules, then we get the \textbf{coefficient exact sequence} of cohomology groups.
    \begin{align*}
        0 \rightarrow H^0(M, A) \rightarrow H^0(M, B) \rightarrow H^0(M, C) \xrightarrow{\delta^0} H^1(M, A)\\
        \rightarrow H^1(M, B) \rightarrow H^1(M,C) \xrightarrow{\delta^1} H^2(M, A) \rightarrow \dots 
    \end{align*}
\end{corollary}

\begin{proof}
    Produce a projective resolution $P_*$ for $M$ before applying $Hom_\Lambda(-, N)$ where $ N = A,B,C$ to give the short exact sequence of cochain complexes: 
    \begin{align*}
        0 \rightarrow Hom(P_*, A) \rightarrow Hom(P_*, B) \rightarrow Hom(P_*, C) \rightarrow 0
    \end{align*}
    Now apply the Mayer-Vietoris theorem to get the long exact sequence in cohomology.
\end{proof}

\begin{lemma}
    Let $B \xrightarrow{\partial} A \xrightarrow{\natural} A/im({\partial}) \rightarrow 0$ be exact, then the induced function upon taking Hom$(-, N)$:
    \begin{align*}
        \natural^* : Hom(A/im(\partial), N) \to ker(\partial^*) \subseteq Hom(A,N)
    \end{align*}
    ss an isomorphism for any $\Lambda$-module $N$
\end{lemma}
\begin{proof}
    Apply contravariant functor Hom$(-, N)$ to the exact sequence to get: 
    \begin{align*}
        Hom(A/im(\partial), N) \xrightarrow{\natural^*} Hom(A, N) \xrightarrow{\partial^*} Hom(B,N)
    \end{align*}
    Let $f \in Hom(A/im(\partial), N)$,  $\partial^*(\natural^*(f)) = (\natural \circ \partial)^*(f) = f \circ (\natural \circ \partial) = 0$ by exactness. So im$(\natural^*) \subseteq ker(\partial^*)$ and the definition makes sense.\\
    Let $\natural^*(f) = 0$. Let $a + im(\partial) \in A/im(\partial)$ then $f(a + im(\partial)) = f(\natural(a)) = \natural^*(f(a)) = 0$. Hence $f = 0$ and $\natural^*$ is injective.\\
    Now let $g \in ker(\partial^*)$. $\partial^*(g) = g \circ \partial = 0$ so for all $b \in B$, $g(\partial(b)) = 0$ hence $im(\partial) \subseteq ker(g)$. Since $g : A \to N$, by the universal property of quotient modules we have a map $\hat{g}: A/im(\partial) \to N$ such that $ g = \hat{g} \circ \natural = \natural^*\hat{g}$. Hence $\natural$ is surjective.
\end{proof}

\begin{proposition}
    Fix some ring $\Lambda$, then $H^0(M, N) \cong Hom_\Lambda(M,N)$
\end{proposition}
\begin{proof}
    Consider the following resolution of $M$: 
    \begin{align*}
        P_1 \xrightarrow{\partial_1} P_0 \xrightarrow{\epsilon} M \rightarrow 0
    \end{align*}
    Since $\epsilon$ is surjective, by the first isomorphism theorem and exactness we have that $P_0/ker(\epsilon) = P_0/im(\partial_1) \cong M$. Now applying the above lemma we get that $\epsilon^* : Hom(M, N) \to ker(\partial_1^*)$ is an isomorphism. The $0$th cohomology group $H^0 = Ker(\partial^*_1)/im(nothing) = Ker(\partial^*_1)$ so the statement is proved. 
\end{proof}

\begin{remark}
    So if we wanted to, we could write the coefficient exact sequence of $\Lambda$-module $M$ given the above short exact sequence as: 
    \begin{align*}
        0 \rightarrow Hom_\Lambda(M,A) \rightarrow Hom_\Lambda(M, B) \rightarrow Hom_\Lambda(M,C) \xrightarrow{\delta^0} H^1(M, A)\\
        \rightarrow H^1(M, B) \rightarrow H^1(M,C) \xrightarrow{\delta^1} H^2(M, A) \rightarrow \dots 
    \end{align*}
    And so the sequence  $0 \rightarrow Hom_\Lambda(M,A) \rightarrow Hom_\Lambda(M, B) \rightarrow Hom_\Lambda(M,C) \rightarrow \dots$ is exact iff $H^1(M,A)$ is zero! This then, shows that the first cohomology group is the first step in measuring how the $Hom$ functor has failed to be exact. 
\end{remark}

\begin{remark}
    NOTE. When we actually want to compute the cohomology $H^k(M,N)$ of a specific module we have the following workflow: 
    \begin{enumerate}
        \item Take a resolution $P \to M$ (either free or projective).
        \item Delete $M$ from $P\to M$ to just get the projective part. 
        \item Apply $Hom(-,N)$ to the unaugmented resolution 
        \item Take Cohomology, paying close attention to the fact that the indices have now been shifted by removing $M$. 
    \end{enumerate}
    If we didn't chop off the $M$, we would get that $H^0(M,N) \cong 0$ rather than $Hom(M,N)$. Leaving $M$ in would essentially be asking how does $M$ relate to itself, rather than measuring the internal structure of $M$ given by the projective resolution. In this sense, the cohomology groups $H^n(M,N)$ are defined as $\ker(\partial^*_{n+1})/im(\partial^*_{n})$ where $\partial^*_{n+1}: Hom(P_{n},N) \to Hom(P_{n+1},N)$. NOTE, this deleting strategy only works when what is left behind is projective. If you just have a short exact sequence you need to take cohomology groups via mayer vietoris and work it out based on what you already know. 
\end{remark}

\subsection{Reverse Exact Sequence}
It turns out there is another natural way in which to get a long exact sequence of cohomology groups from some short exact sequence
\begin{align*}
    0 \rightarrow A \xrightarrow{f} B \xrightarrow{g} C \rightarrow 0 
\end{align*}
and a $\Lambda$-module $M$. Instead of taking a projective resolution of $M$, we take projective resolutions of $A,B$, and $C$, make them into a short exact sequence and then apply the \textit{contravariant} functor $Hom_\Lambda(-,M)$. This then induces the \textbf{reverse exact sequence} of cohomology groups. The proofs of each proposition leading up to applying Mayer-Vietoris requires doing lots of finnicky algebra and huge diagrams so I've made the inspired decision to leave it out of these notes, sorry!

\begin{remark}
    If we take any resolution $P_* \rightarrow A$, $T_* \rightarrow B$, and $Q_* \rightarrow C$, then the sequence
    \begin{align*}
        P_* \rightarrow T_* \rightarrow Q_*
    \end{align*}    
    is \textbf{not} necessarily exact! However, it turns out that we can construct a resolution for $B$ in such a way that the sequence does in fact become exact.
\end{remark}

\begin{theorem}
    Let $0 \rightarrow A \xrightarrow{f} B \xrightarrow{g} C \rightarrow 0$ be an exact sequence of $\Lambda$-modules, $P_* \rightarrow A$ and $Q_* \rightarrow C$ be projective resolutions for A and C respectively. Then there exists a projective resolution $T_* \rightarrow B$ for $B$ such that for all $n$, $T_n = P_n \oplus Q_n$ and the map between each $T_n$ is given by 
    \begin{align*}
        \partial^T_n = 
        \begin{pmatrix}
            \partial^P_n & ? \\
            0            & \partial^Q_n
        \end{pmatrix} 
    \end{align*} 
\end{theorem}

\begin{proposition}
    Let $P_*$, $Q_*$ and $T_*$ be as above. Then the sequence 
    \begin{align*}
        0 \rightarrow P_* \rightarrow T_* \rightarrow Q_* \rightarrow 0
    \end{align*}
    is a short exact sequence. 
\end{proposition}

\begin{lemma}
    Let $0 \rightarrow P \xrightarrow{i} P \oplus Q \xrightarrow{\pi} Q \rightarrow 0$ be a split exact sequence of $\Lambda$-modules. Then for any $\Lambda$-module $M$, we have that 
    \begin{align*}
        0 \rightarrow Hom(Q,M) \xrightarrow{\pi^*} Hom(P \oplus Q,M) \xrightarrow{i^*} Hom(P,N) \rightarrow 0
    \end{align*}
    Is also split exact. 
\end{lemma}

\begin{corollary}
    Let $0 \rightarrow P_* \rightarrow T_* \rightarrow Q_* \rightarrow 0$ be a short exacts sequence of cochain complexes as described above, then the sequence
    \begin{align*}
        0 \rightarrow Hom(Q_*,M) \rightarrow Hom(T_*,M) \rightarrow Hom(P_*,N) \rightarrow 0    
    \end{align*}
    is a short exact sequence. 
\end{corollary}

\begin{theorem}(Reverse Exact Sequence)
    Let $0 \rightarrow A \xrightarrow{i} B \xrightarrow{p} C \rightarrow 0$ be a short exact sequence of $\Lambda$-modules. Then for any $\Lambda$-module $M$, we get a long exact sequence of cohomology groups
    \begin{align*}
        0 \rightarrow H^0(C,M) \xrightarrow{p^*} H^0(B,M) \xrightarrow{i^*} H^0{C,M} \xrightarrow{\delta}\\
        H^1(C,M) \xrightarrow{p^*} H^1(B,M) \xrightarrow{i^*} H^1{C,M} \xrightarrow{\delta} \dots \\
        \dots \xrightarrow{\delta} H^n(C,M) \rightarrow H^n(B,M) \rightarrow H^n(A,M) \rightarrow \dots 
    \end{align*}
    known as the \textbf{reverse exact sequence}. Note the contravariant nature. 
\end{theorem}

\subsection{Baer's Theorem}

\begin{definition}
    Fix two $\Lambda$-modules $M$ and $N$. We define $\mathcal{E}xt(M,N)$ to be the set of all extensions of $M$ by $N$, that is all exact sequences of the form: 
    \begin{align*}
        0 \rightarrow N \rightarrow A \rightarrow M \rightarrow 0
    \end{align*}
    for some $\Lambda$-module $A$. 
\end{definition}

We can then define an equivalence relation on this set. 

\begin{definition}
    Two extensions $A$ and $A'$ in $\mathcal{E}xt(M,N)$ are said to be equivalent (and we write $A \equiv A'$) if there is a commutative diagram
        \[
        \begin{tikzcd}
        0 \arrow[r] & N \arrow[r] \arrow[d, equal] 
        & A \arrow[r] \arrow[d, "\cong"] 
        & M \arrow[r] \arrow[d, equal] 
        & 0 \\
        0 \arrow[r] & N \arrow[r] 
        & A' \arrow[r] 
        & M \arrow[r] 
        & 0
        \end{tikzcd}
        \]
    Where the map from $A$ to $A'$ is an isomorphism by the five lemma. It is easy to check that this is an equivalence relation.   
\end{definition}

\begin{definition}
    We define $Ext(M,N)$ to be the equivalence classes of $\mathcal{E}xt(M,N)$ that is: 
    \begin{align*}
        Ext(M,N) := \mathcal{E}xt(M,N) / \equiv 
    \end{align*}
\end{definition}

\begin{theorem}(Baer's Theorem)
    Fix $\Lambda$-modules $M$ and $N$. Then
    \begin{enumerate}
        \item $Ext(M,N)$ is an additive abelian group 
        \item $Ext(M,N) \cong H^1(M,N)$
    \end{enumerate}
\end{theorem}
The proof of this is a little ridiculous so I'll cast most of the details to the appendix. However, I will sketch out the main steps and expand on them a little below:
\begin{enumerate}
    \item Given an exact sequence $0 \rightarrow K \xrightarrow{i} X \xrightarrow{p} M \rightarrow M$, define the colimit function $\lim_{\rightarrow} : Hom_R(K,N) \to Ext(M,N)$ given by $f \mapsto (0 \rightarrow N \xrightarrow{j} \lim_{\rightarrow}(f,i) \xrightarrow{\pi} M \rightarrow 0)$ as defined by the pushout.
    \item Prove that $\lim_{\rightarrow}$ is surjectice when $X$ is a projective module. 
    \item Equip $Ext(M,N)$ with addition: Define the limit function $\lim_{\leftarrow}: Hom_R(K,M) \to Ext_R(K,N)$ where $f \mapsto 0 \rightarrow N \xrightarrow{\iota} \lim_{\leftarrow}(f,p) \xrightarrow{\pi_K} K \rightarrow 0$ as defined by the pullback. Let $\Delta: M \to M \oplus M$ be the diagonal map and define addition by composing this delta and regular addition with the pullback limit function. 
    \item Check that $\lim_{\rightarrow}$ preserves the aforementioned addition. As then $\lim_{\rightarrow}$ is a surjective magma homomorphism from an abelian group to $Ext(M,N)$ implying that $Ext(M,N)$ is abelian. 
    \item Prove that the identity of $Ext(M,N)$ is the trivial split exact sequence. 
    \item Apply Mayer Vietoris to $0 \rightarrow K \xrightarrow{i} X \xrightarrow{p} M \rightarrow M$ to prove $Ext(M,N) \cong H^1(M,N)$
\end{enumerate}

\begin{lemma}
    The trivial split extension is the identity element of $Ext(M,N)$. 
\end{lemma}
\begin{proof}
    Suppose that we have the follow diagram: 
    \[
    \begin{tikzcd}
        0 \arrow[r] & K \arrow[r, "i"] \arrow[d, "\mathbf{0}"] & X \arrow[r, "p"] \arrow[d, "\nu"] & M \arrow[r] \arrow[d, equal] & 0\\
        0 \arrow[r] & N \arrow[r, "j"] & \lim_\rightarrow (\mathbf{0}, i) \arrow[r, "\pi"] & M \arrow[r] & 0
    \end{tikzcd}
    \]
    Observe that $\lim(\mathbf{0}, i) = (N \oplus X)/im(\mathbf{0} \times i) \cong N \oplus X/im(i)$, which by exactness gives $N \oplus X/im(i) = N \oplus X/\ker(p) \cong M$ by the first isomorphism theorem. Hence $\lim_\rightarrow(\mathbf{0}, i) \cong N \oplus M$. So $\lim_{\rightarrow}(\mathbf{0}) = [T]$ where $[T]$ is the equivalence class given by the trivial split extension. 
\end{proof}

\begin{theorem}(Second Part of Baer's)
    $Ext(M,N) \cong H^1(M,N)$ 
\end{theorem}
\begin{proof}
    If $T \in \mathcal{E}xt(M,N)$ is the trivial extension, it is straightforward to see that $[T] = \lim_{\rightarrow}(\mathbf{0})$ where $\mathbf{0} : K \to N$ denotes the zero homomorphism.
    If $f \in Hom(K,N)$ has the property that $\lim_{\rightarrow}(f) = [T]$ then by \ref{TrivPushoutExtension} $\exists \hat{f}: P \to N$ making the below diagram commute:
    \[
    \begin{tikzcd}
        K \arrow[d, "f"] \arrow[r, "i"] & P \arrow[ld, "\hat{f}"]\\
        N
    \end{tikzcd}
    \]
    Moreover it follows from \ref{Magma} that $Ext(M,N) \cong Hom(K,N)/(\lim^{-1}([T]))$ so we get the exact sequence: 
    \begin{align*}
        Hom(P,N) \xrightarrow{i^*}  Hom(K,N) \xrightarrow{\lim_{\rightarrow}} Ext(M,N) \rightarrow 0
    \end{align*}
    But also, by the reverse exact sequence applied to our original SES $0 \rightarrow K \xrightarrow{i} P \xrightarrow{p} M \rightarrow 0$, we get 
    \begin{align*}
        Hom(P,N) \xrightarrow{i^*} Hom(K,N) \xrightarrow{\delta} H^1(M,N) \rightarrow 0
    \end{align*}
    So $Ext(M,N) \cong H^1(M,N)$. 
\end{proof}


This is arguably one of the most important theorems in this course. This is to say that if we can compute the first cohomology group, then we can compute exact sequences. In other words, as stated before, $H^1$ truly measures the failure of the $Hom$ functor to be exact. Without this theorem, if someone asked you to find all possible modules A such that $0 \rightarrow N \rightarrow A\rightarrow M\rightarrow 0$ is exact, you would have to guess-and-check every possible module combination.

Because of Baer's Theorem, you don't have to search for modules at all; you just use your "machinery" (like projective resolutions) to calculate the group $H^1$. The group gives you the exact number and structure of those possible sequences before you even start looking for them.

\begin{example}
    \begin{enumerate}
        \item Let $\Lambda = \mathbb{Z}$, and $M = N = \mathbb{Z}/3$. We want to compute $H^1(\mathbb{Z}/3, \mathbb{Z}/3)$; in other words calculate the number and structure of extensions of $\mathbb{Z}/3$ by $\mathbb{Z}/3$.\\
        Start by taking a free resolution of $\mathbb{Z}/3$ over $\mathbb{Z}$: 
        \begin{align*}
            0 \rightarrow \mathbb{Z} \xrightarrow{\times 3} \mathbb{Z} \xrightarrow{\epsilon} \mathbb{Z/3} \rightarrow 0
        \end{align*}
        Take $Hom(-, \mathbb{Z}/3)$ of this exact sequence, droping the first term\footnote{We do this because we are interested in the cohomology of the projective resolution and thus the failure of exactness, not the cohomology of $\mathbb{Z}/3$ itself}, noting that this is contravariant. We get
        \begin{align*}
            0 \rightarrow Hom_\mathbb{Z}(\mathbb{Z}, \mathbb{Z}/3) \xrightarrow{(\times 3)^*} Hom_\mathbb{Z}(\mathbb{Z}, \mathbb{Z}/3) \rightarrow{0}
        \end{align*}
        Note that $(\times 3)^* = \psi \circ \times 3$ where $\psi$ is the group hom from $Hom_\mathbb{Z}(\mathbb{Z}, \mathbb{Z}/3) \to Hom_\mathbb{Z}(\mathbb{Z}, \mathbb{Z}/3)$. Hence subbing $\psi$ as the identity gives (and applying \ref{HomProp1}) we get 
        \begin{align*}
            0 \rightarrow \mathbb{Z}/3 \xrightarrow{\times 3} \mathbb{Z}/3 \rightarrow 0
        \end{align*}
        Now, $H^1 = \frac{ker(\mathbb{Z}/3 \to 0)}{im(\times 3 mod 3)} = \mathbb{Z}/3$
        So $H^1(\mathbb{Z}/3, \mathbb{Z}/3) = \mathbb{Z}/3 = Ext(\mathbb{Z}/3, \mathbb{Z}/3)$
    \end{enumerate}
\end{example}

\subsection{More Details on Projective Modules}
Now that we have the required machinery, we can give some more details on projective modules. 

\begin{proposition}
    Let $P$ be a $\Lambda$-module. Then $P$ is projective iff $H^1(P,N) = 0$ for all $\Lambda$-modules $N$.    
\end{proposition}
\begin{proof}
    If $P$ is projective, consider the following projective resolution of $P$: $0 \rightarrow P \xrightarrow{id} P \rightarrow 0$. Taking the coefficient exact sequence immediately gives that $H^1(P,N) = 0$ for all modules $N$. 
    \\Conversely, assume that $H^1(P,N) = 0$ for all $\Lambda$-modules $N$. Let $(e_i)_{i \in I}$ be a spanning set for $P$ and F be the free module with $(E_i)_{i \in I}$ such that $\pi(E_i) = e_i$. Setting $K = \text{ker}(\pi)$ we get the exact sequence: 
    \begin{align*}
        0 \rightarrow K \xrightarrow{i} F \xrightarrow{\pi} P \rightarrow 0
    \end{align*}
    where $i$ is the inclusion map. Take the coefficient exact sequence (i.e. $H^*(P, -)$) to get 
    \begin{align*}
        0 \rightarrow Hom(P,K) \xrightarrow{i_*} Hom(P,f) \xrightarrow{\pi_*} Hom(P,P) \rightarrow H^1(P,K) \rightarrow \dots 
    \end{align*}
    which since $H^1(P,N) = 0$ for any $N$ we have: 
    \begin{align*}
        0 \rightarrow Hom(P,K) \xrightarrow{i_*} Hom(P,F) \xrightarrow{\pi_*} Hom(P,P) \rightarrow 0
    \end{align*}
    is exact. Therefore $\pi_*$ is surjective, so there exists some $s \in Hom(P,F)$ such that $\pi_*(s) = \pi \circ s = id_P$. This implies that the original sequence splits on the right and therefore splits. Then $F \cong P \oplus K$ where $F$ is free thus $P$ is projective by the summand definition. 
\end{proof}

This implies that projective modules are effectively "invisible" to cohomology. We can also prove the above using the reverse exact sequence, which I will include below just for a bit of fun. Note that both the above and below arguments can also be used to show that $P$ is projective iff $H^n(P,N) = 0$ for all $n \in \mathbb{N}$

\begin{proposition}
    (Reverse Exact Sequence Version) Let $P$ be a $\Lambda$-module. Then $P$ is projective iff $H^1(P,N) = 0$ for all $\Lambda$-modules $N$.
\end{proposition}

\begin{proposition} (Schanuel's Lemma)
    Let $0 \rightarrow I \xrightarrow{i} P \xrightarrow{p} M \rightarrow 0$ and $0 \rightarrow J \xrightarrow{j} Q \xrightarrow{q} M \rightarrow 0$ be short exact sequences of $\Lambda$-modules wherein $P, Q$ are projective. Then $I \oplus Q \cong J \oplus P$.
\end{proposition}
\begin{proof}
    Define the pullback module $\lim_{\leftarrow}(p,q) = \{(x,y) \in P \oplus Q: p(x) = q(y)\}$. Define the map $\pi_1 : \lim_{\leftarrow}(p,q) \to P$ as the projection of the pullback onto $P$. Pick any element $x \in P$ and consider its image in $M$, $p(x)$. Since $q$ is surjective, we have that there must exist a $y \in Q$ such that $p(x) = q(y)$, therefore $(x,y) \in \lim_{\leftarrow}(p,q)$. Since $\pi_1(x,y) = x$ this means that $\pi_1$ is surjective. Similarly, we can show the projection $\pi_2: \lim_{\leftarrow} \to Q$ is surjective. So there we get two exact sequences: 
    \begin{align*}
        0 \rightarrow \ker{\pi_1} \rightarrow \lim_{\leftarrow}(p,q) \xrightarrow{\pi_1} P \rightarrow 0\\
        0 \rightarrow \ker{\pi_2} \rightarrow \lim_{\leftarrow}(p,q) \xrightarrow{\pi_2} Q \rightarrow 0
    \end{align*}
    Observe: 
    \begin{align*}
        \ker(\pi_1) &= \{(x,y) \in \lim_{\leftarrow}(p,q): p(x) = q(y) \land x = 0 \}\\
                    &= \{(0,y) \in \lim_{\leftarrow}(p,q): q(y) = 0\}\\
                    &\cong \ker(q) = J                   
    \end{align*}
    So we get two exact sequences by symmetry:
    \begin{align*}
        0 \rightarrow J \rightarrow \lim_{\leftarrow}(p,q) \xrightarrow{\pi_1} P \rightarrow 0\\
        0 \rightarrow I \rightarrow \lim_{\leftarrow}(p,q) \xrightarrow{\pi_2} Q \rightarrow 0
    \end{align*}
    Since $P$ and $Q$ are projective, both sequenes split hence $\lim_{\leftarrow}(p,q) \cong P \oplus J$ and $\lim_{\leftarrow}(p,q) = Q \oplus I$
\end{proof}

\begin{proposition}(Dimension Shift) \label{DimShift}
    Let $0 \rightarrow K \rightarrow P_0 \rightarrow M \rightarrow 0$ be an exact sequence of $\Lambda$-modules. If $P_0$ is projective then $H^n(K,N) \cong H^{n+1}(M,N)$ for all $\Lambda$-modules $N$. 
\end{proposition}
\begin{proof}
    Apply the contravariant functor $Hom(-, N)$ and apply the mayer-vietoris theorem to get a long exact sequence in cohomology: 
    \begin{align*}
        \dots \rightarrow H^n(M,N) \rightarrow H^n(P_0, N) \rightarrow H^n(K,N) \xrightarrow{\delta} H^{n+1}(M,N) \rightarrow H^{n+1}(P_0, N) \rightarrow \dots
    \end{align*}
    for a particular $n \leq 1$. But $H^n(P_0, N) = 0$ for all $n \geq 1$ since it is projective, so for a particular $n \geq 1$ we have: 
    \begin{align*}
        0 \rightarrow H^n(K,N) \xrightarrow{\delta} H^{n+1}(M,N) \rightarrow 0
    \end{align*}
    Which implies that $\delta$ is an isomorphism, so $H^n(K,N) \cong H^{n+1}(M,N)$.  
\end{proof}

\begin{remark}
    Suppose that we have some long exact sequence 
    \begin{align*}
        0 \rightarrow P_n \xrightarrow{\partial^{n}} P_{n-1} \xrightarrow{\partial^{n-1}} \dots \xrightarrow{\partial^1} P_0 \xrightarrow{\partial^0} M \rightarrow 0
    \end{align*}
    Define\footnote{Note that these are called the nth syzygies of $M$, from the greek syzygos meaning "yoked together"} $K_n = \ker{\partial^n}$, then we get a family of short exact sequences as follows. Clearly for $n = 0$
    \begin{align*}
        0 \rightarrow K_0 \rightarrow P_0 \xrightarrow{\partial^0} M \rightarrow 0 
    \end{align*}
    is exact by definition. Then for $n=1$:
    \begin{align*}
        0 \rightarrow K_1 \rightarrow P_1 \xrightarrow{\partial^1} K_0 \rightarrow 0 
    \end{align*}
    is exact since $im(\partial^1) = \ker(\partial^0) = K_0$ by exactness, so $\partial^1$ is surjective. We can continue this pattern up to: 
    \begin{align*}
        0 \rightarrow K_n \rightarrow P_n \xrightarrow{\partial^{n}} K_{n-1} \rightarrow 0 
    \end{align*}
    But $\partial^n$ is injective, so $K_n = 0$ and thus $P_n \cong K_{n-1}$. 
\end{remark}

\begin{proposition}
    Suppose we have a long exact sequence
    \begin{align*}
        0 \rightarrow K \xrightarrow{\partial^{n}} P_{n-1} \xrightarrow{\partial^{n-1}} \dots \xrightarrow{\partial^1} P_0 \xrightarrow{\partial^0} M \rightarrow 0
    \end{align*}
    of $\Lambda$-modules. Then $K$ is projective iff $H^{n+1}(M,N) = 0$ for all $n \in \mathbb{N}$ and all coefficient modules $N$. 
\end{proposition}
\begin{proof}
    From the above remark with $P_n := K$ we get a family of short exact sequences. 
    \begin{align*}
        0 \rightarrow K_0 \rightarrow &P_0 \rightarrow M \rightarrow 0 \\
        0 \rightarrow K_1 \rightarrow &P_1 \rightarrow K_0 \rightarrow 0 \\
        &\; \vdots\\
        0 \rightarrow K \rightarrow &P_{n-1} \rightarrow K_{n-1}
    \end{align*}
    Then from repeated applications of Proposition \ref{DimShift} we get: 
    \begin{align*}
        H^{n+1}(M,N) &\cong H^n(K_0,N)\\
        H^n(K_0, N)  &\cong H^{n-1}(K_1, N)\\
                     &\; \vdots\\
        H^2(K_{n-1}, N) & \cong H^1(K,N) 
    \end{align*}
    so $H^1(K,N) \cong H^{n+1}(M,N)$ which is zero iff $K$ is projective. 
\end{proof}   


\begin{proposition}
    $P$ and $Q$ are projective iff $P \oplus Q$ are projective. 
\end{proposition}
\begin{proof}
    Assume that $P$ and $Q$ are projective, then by the summand definition there exists modules $K$ and $K'$ such that $F \cong P \oplus K$ and $F' \cong Q \oplus K'$ where $F$ and $F'$ are free modules. Then $F \oplus F' = (P \oplus K) \oplus (Q \oplus K') = (P \oplus Q) \oplus (K \oplus K')$. Since the sum of free modules is still free (easy to show) $P \oplus Q$ is a summand of a free module and is thus projective. \\
    Conversely assume that $P \oplus Q$ is projective. Then there exists a module $K$ such that $F \cong (P \oplus Q) \oplus K$ where $F$ is free. Then by associativity we have $F \cong P \oplus (Q \oplus K)  = Q \oplus (P \oplus K)$. Hence $P$ and $Q$ are both projective by the summand definition.  
\end{proof}

\begin{lemma}
(Whitehead's Lemma) Let $0 \rightarrow P_n \rightarrow P_{n-1} \rightarrow \dots \rightarrow P_1 \rightarrow P_0 \rightarrow 0$ be a long exact sequence of projective modules, then we have: 
\begin{align*}
    \bigoplus_{r \geq 0} P_{2r} \cong \bigoplus_{s \geq 0} P_{2s+1}
\end{align*}
\end{lemma}

\section{Projective and Cohomological Dimension}
\begin{definition}
    Fix a ring $\Lambda$ and $\Lambda$-module $M$, we define the cohomological dimension of $M$: 
    \begin{align*}
        cd(M) = \text{min}\{n : \forall N, H^{n+1}(M,N) = 0\}
    \end{align*}
    If no such $n$, then $cd(M) = \infty$.    
\end{definition}

\begin{definition}
    For a fixed $\Lambda$ and $\Lambda$-module $M$. We define the projective dimension: 
    \begin{align*}
        pd(M) \leq n \iff \exists \text{ a projective resolution for $M$ of length n}
    \end{align*}
    i.e. We have a projective resolution for $M$: 
    \begin{align*}
        0 \rightarrow P_n \rightarrow P_{n-1} \rightarrow \dots \rightarrow P_1 \rightarrow P_0 \rightarrow M
    \end{align*}
    If no such $n$ exists, then $pd(M) = \infty$
\end{definition}

\begin{proposition}
    If $pd(M) \leq n$ then $cd(M) \leq n$
\end{proposition}
\begin{proof}
    Take a projective res of $M$, take $Hom_\Lambda(-,N)$. Then $H^{n+1} = 0$ for all $N$. 
\end{proof}

We can get the converse via dimension shifting, so therefore we have: 

\begin{proposition}
     $pd(M) \leq n$ iff $cd(M) \leq n$.
\end{proposition}
\begin{proof}
    Forward direction proved above. For backwards direction take a projective resolution of $M$ 
    \begin{align*}
        \dots \rightarrow P_n \xrightarrow{\partial_n} P_{n-1} \xrightarrow{\partial_{n-1}} \dots \xrightarrow{\partial_1} P_0 \xrightarrow{\partial_0} M \rightarrow 0
    \end{align*}
    Defining $K = \ker(\partial_{n-1}) = im(\partial_n)$ gives us an exact sequence.
    \begin{align*}
        0 \rightarrow K  \xrightarrow{\partial_n} P_{n-1} \xrightarrow{\partial_{n-1}} \dots \xrightarrow{\partial_1} P_0 \xrightarrow{\partial_0} M \rightarrow 0
    \end{align*}
    Then using the dimension shifting argument from \ref{DimShift}, we have that $H^{n+1}(M,N) = H^1(K,N)$. If cd($M$) $\leq n$ this implies that $H^{n+1}(M,N) = 0 = H^1(K,N)$. This then means that $K$ is projective so the above sequence is a projective resolution of length $n$, so pd($M$) $\leq n$. 
\end{proof}

\begin{definition}
    Let $\Lambda$ be a ring. We say that the global dimension of $\Lambda$ $gd(\Lambda) \leq n \iff H^{n+1}(M,N) = 0$ for all $\Lambda$-modules $M,N$. That is to say $cd(M) \leq n$ for all $\Lambda$-modules $M$. Moreover, every modules $M$ has projective resolution of length at most $n$. 
\end{definition}

\section{Noetherian Rings and Modules}

This section will go into detail on what are known as noetherian rings and modules. Although there are a few characterisations, the intuitive way to understand them is that they have a "general finiteness" condition, allowing us to treat complex algebraic objects as essentially finite. Before the definition of a noetherian ring was formalised, Hilbert characterised them as being "not too large". 

\begin{definition}
    Let $\Lambda$ be a ring, and $M$ a $\Lambda$-module. $M$ is \textbf{noetherian} if when given a sequences of submodules $(M_r)_{r \in \mathbb{N}}$ such that $M_r \subsetneq M_{r+1}$ then $\exists n$ such that $M_r = M_n$ for all $r >n$. I.e. the ascending chain terminates. 
\end{definition}

It is important to note that there is an equivalent definition to noetherian modules in that $M$ is noetherian if every submodule of $M$ is finitely generated. Showing that our definition above and this definition is equivalent requires assuming the axiom of choice, which for the most part, we are not doing in this course!

\begin{proposition}
    If $M$ is noetherian, then $M$ is finitely generated over $\Lambda$.     
\end{proposition}
\begin{proof}
    If $\{e_1, \dots, e_n\} \subset M$, define $[e_1, \dots, e_n) = \{\sum^n_{i=1}e_i \lambda_i : \lambda_i \in \Lambda\}$ as the submodule genereated by those elements.
    Choose some $e_1 \in M$. If $M = [e_1)$, then we are done. Otherwise choose $e_2 \in M\setminus [e_1)$. Again, if $M = [e_1, e_2)$ then we are done.\\ Suppose that we continue this process $n$ elements $\{e_1, \dots, e_n\} \subset M$ where $[e_1, \dots, e_n) \neq M$. Then choose $e_{n+1} \in M \setminus [e_1, \dots, e_n)$. \\Suppose we can continue this indefinitely, then if we define $M_r = [e_1, \dots, e_r)$, we get an indefinite ascending chain of submodules $M_r \subset M_{r+1}$. $M$ is noetherian so this must terminate, i.e. $\exists n$ s.t. $M = [e_1, \dots, e_n)$
\end{proof}

\begin{proposition}
    If $M$ is noetherian, any $\Lambda$-submodule of $M$ is noetherian. 
\end{proposition}
\begin{proof}
    Let $K \subset M$ be a submodule of $M$. Take an ascending chain of submodules inside $K$: $K_1 \subset K_2 \subset K_3 \subset \dots$. Each $K_i$ is also submodule of $M$ and so by the definition of $M$ being noetherian this chain must terminate. Hence $K$ is also noetherian since any ascending chain of submodules terminates. 
\end{proof}

\begin{proposition}
    Let $p : M \to Q$ be a surjective $\Lambda$-homomorphism. If $M$ is noetherian then $Q$ is noetherian.  
\end{proposition}
\begin{proof}
    Let $(Q_r)_{0 \leq r}$  be an ascending sequence of $\Lambda$-modules in $Q$. Then since $p$ is surjective, $p^{-1}(Q_r)$ is an ascending sequence of $\Lambda$-modules in $M$.\\
    $M$ is noetherian, so $\exists N$ such that $\forall r \geq N$, $p^{-1}(Q_r) = p^{-1}(Q_N)$. Since $p$ is surjective, applying $p$ to both sides of this equations gives $Q_r = Q_N$ for all $r \geq N$. Hence $Q$ is noetherian. 
\end{proof}

Now we actually get to use a bit of our homological stuff! 

\begin{proposition}
    Let $0 \rightarrow K \xrightarrow{i} M \xrightarrow{p} Q \rightarrow 0$ be an SES of of $\Lambda$-modules. $K$ and $Q$ are noetherian iff $M$ is noetherian.  
\end{proposition}
\begin{proof}
    Assume that $K$ and $Q$ are noetherian. Let $N \subseteq M$ be a submodule of $M$. $K \cap N$ is a submodule of $K$, and since $K$ is noetherian, $K \cap N$ is finitely generated. Similarly, $p(N)$ is a submodule of $Q$, and thus is also finitely generated as $Q$ is noetherian. Take a generating set for both, noting that the sequence: 
    \begin{align*}
        0 \rightarrow K \cap N \rightarrow N \rightarrow p(N) \rightarrow 0 
    \end{align*}
    is exact. Then the union of each generating set generates $N$. \\
    Now assume that $M$ is noetherian. $K$ is clearly noetherian since it is a submodule of $M$. $Q$ is also noetherian since $p$ is a surjective homomorphism. 
\end{proof}

This leads to one of the most famous theorems in the history of mathematics

\begin{theorem}
    (Hilbert's Basis Theorem 1890) Let $R$ be a commutative Noetherian ring. Then $R[t]$ is also noetherian. 
\end{theorem}

I won't (at the moment) include the proof for this theorem, but it was noted as being one of the first proofs to use non-constructive methods. Although this is commonplace nowadays, it caused a great shock in the mathematical community when first published, eliciting reactions such as "This is not mathematics, it is theology" from Paul Gordan. We will end this section with a few more propositions about noetherian modules. 

\begin{proposition}
    Suppose that $A$ is an $R$-module, and $\eta : A \to A$ is surjective but not injective. Then $A$ is not noetherian. 
\end{proposition}
\begin{proof}
Let $K = \ker \eta \neq \{0\}$. By the First Isomorphism Theorem, there exists an isomorphism $\psi: A/K \to A$. Define a sequence of submodules $\{B_r\}_{r \in \mathbb{N}}$ inductively:
\begin{itemize}
    \item $B_0 = \{0\}$
    \item $B_{r+1} = \pi^{-1}(\psi^{-1}(B_r))$, where $\pi: A \to A/K$ is the canonical projection.
\end{itemize}
This implies $B_{r+1}/K = \psi^{-1}(B_r)$. We show by induction that $B_{r+1}/B_r \cong K$ for all $r \geq 0$:
\begin{enumerate}
    \item \textbf{Base case ($r=0$):} $B_1/B_0 = K/\{0\} \cong K$.
    \item \textbf{Inductive step:} Assume $B_r/B_{r-1} \cong K$. By the Third Isomorphism Theorem:
    \[ \frac{B_{r+1}}{B_r} \cong \frac{B_{r+1}/K}{B_r/K} = \frac{\psi^{-1}(B_r)}{\psi^{-1}(B_{r-1})} \]
    Since $\psi^{-1}$ is an isomorphism, it preserves quotients:
    \[ \frac{\psi^{-1}(B_r)}{\psi^{-1}(B_{r-1})} \cong \frac{B_r}{B_{r-1}} \cong K \]
\end{enumerate}
Since $K \neq \{0\}$, the chain $B_0 \subsetneq B_1 \subsetneq B_2 \subsetneq \dots$ is strictly increasing. Thus, $A$ is not Noetherian.
\end{proof}

\begin{proposition}
    Let $A$ and $S$ be $\Lambda$-modules such that $A \oplus S \cong A$. If $A$ is noetherian show that $S = 0$
\end{proposition}
\begin{proof}
    Contraposing this we want to show that if $S$ is nonempty then $A$ is non noetherian. We have the sequence $A \xrightarrow{\phi} A \oplus S \xrightarrow{\pi} A$ where $\phi$ is the isomorphism, and $\pi$ is the projection of $A \oplus S$ onto $A$ by sending all elements of $S$ to 0. Then $\pi \circ \phi : A \to A$ is surjective as the composition of surjective maps but not injective since $S \neq 0$, hence by the previous theorem, $A$ is non-noetherian. 
\end{proof}

\begin{proof}
    Let $\Lambda$ be a right noetherian ring. If $m,n$ are positive integers such that $\Lambda^m \cong \Lambda^n$ show that $m = n$.
\end{proof}
\begin{proof}
    WLOG $m < n$, then $\Lambda^m \cong \Lambda^m \oplus \Lambda^{n-m}$. $\Lambda^{n-m}$ is non empty, so by the previous proposition $\Lambda$ is non noetherian, which is a contradiction. Same works for the other case. Hence $m = n$. 
\end{proof}

\begin{proposition}
    Let $R$ be a commutative noetherian ring. If every $R$-module is free, show that $R$ is a field. 
\end{proposition}
\begin{proof}
    Suppose that $a \in R$ is not invertible. Then $(a) \neq R$ and in particular $R/(a) \neq 0$. We thus have the exact sequence: 
    \begin{align*}
        0 \rightarrow (a) \rightarrow R \rightarrow R/(a) \rightarrow 0 
    \end{align*}
    By hypothesis, $(a) \cong R^m$ and $R/(a) = R^n$ for some $n, m$ since they're free. In particular since $R/(a)$ is free, it is projective, so the sequence splits, and we have $R = R^n \oplus R^m$, but $n + m > 1$ so this is impossible. Hence $a$ must be invertible. 
\end{proof}

\section{Classification of Modules over PIDs}
In this section, we will classify all finitely generated modules oversome commutative principal ideal domain. We will actually find out there are only 3 options! But first we will do a bit of recap. 
\subsection{Principal Ideal Domains (PIDs)}
\begin{definition}
    Let $R$ be a commutative ring. $R$ is a \textbf{principal ideal ring} if every ideal in $R$ (i.e. submodule, of the module $R$ over itself) is generated by a single element. $R$ is a \textbf{principal ideal domain} if it is also an integral domain\footnote{no nonzero zero divisors}
\end{definition}

\begin{example}
    Examples of PIDs include 
    \begin{itemize}
        \item The integers $\mathbb{Z}$, through euclidean division. 
        \item Fields ($\mathbb{Q}, \mathbb{R}, \mathbb{C}$)
        \item Polynomial rings over a field $k$, $k[x]$, also through euclidean division. 
        \item Gaussian integers, $\mathbb{Z}[i]$
    \end{itemize}
    Some examples of PIRs which are NOT PIDs include: 
    \begin{itemize}
        \item Integers Mod $n$ where $n$ is a composite number. 
        \item Finite products of PIDs. 
    \end{itemize}
    Some examples of rings which are NOT either include: 
    \begin{itemize}
        \item $\mathbb{Z}[x]$ since the ideal $(2,x)$ is not principle. 
        \item $F[x,y]$, even if $F$ is a field is not a PIR since $(x,y)$ cannot be generated by a single element. 
    \end{itemize}
\end{example}

\begin{proposition}
    If $R$ is a PIR, then $R$ is noetherian. 
\end{proposition}
\begin{proof}
    Every ideal is generated by a single element, thus is finitely generated.
\end{proof}

\begin{corollary}
    If $R$ is a PIR, then $R[t_1, \dots, t_n]$ is noetherian. 
\end{corollary}
\begin{proof}
    Since $R$ is a PIR, by the above proposition, it is noetherian. Then by Hilbert's Basis Theorem, $R[t]$ is noetherian. Simply iterate this process. 
\end{proof}

Beware! $R[t_1, t_2, \dots]$ is not noetherian. 

\subsection{Torsion}
For this entire section, assume that $R$ is a commutative PID. 
\begin{definition}
    An element $m$ of an $R$-module $M$ is called a \textbf{torsion element} of the module if $\exists \lambda \in R$ such that $m\lambda = 0$. The set of all such torsion elements is called the torsion submodule and is denoted $\Theta(M)$. 
\end{definition}

\begin{proposition}
    $\Theta(M)$ is, in fact, a submodule of $M$. 
\end{proposition}
\begin{proof}
    Immediate check of submodule axioms. 
\end{proof}

\begin{definition}
    We say that $M$ is \textbf{torsion free} when $\Theta(M) = 0$ 
\end{definition}
\begin{proposition}\label{TorsionFreeField}
    Let $R$ be a PID. Every $R$-module is torsion free $\iff$ $R$ is a field. 
\end{proposition}
\begin{proof}
    ($\implies$)
    Let $a \in R\setminus {0}$ and $M := R/(a)$. Notice that $a$ is a torsion element of $M$, so for $M$ to be torsion free we need $M = R/(a) = 0$ i.e. $(a) = R$. So $a$ is a unit and thus is invertible. Hence $R$ is a field.\\
    ($\impliedby$) Let $M$ be an $R$-module and moreover let $m \in M$ be a torsion element of $M$. Then $\exists r \in R$ such that $r \cdot m = 0$. Since $R$ is a field, $\exists r^{-1}$. multiplying both sides gives $m = 0$. Hence any torsion element of $M$ must be zero. 
\end{proof}

\begin{proposition}
    $\Theta (M/\Theta(M)) = 0$. i.e. $M/\Theta(M)$ is torsion-free. 
\end{proposition}
\begin{proof}
Let $\pi: M \to M/\Theta(M)$ be the standard quotient map, where we denote the image of an element $m \in M$ as $[m] = m + \Theta(M)$. 

Suppose $[m] \in \Theta(M/\Theta(M))$. By the definition of a torsion element, there exists some non-zero scalar $\lambda \in R$ ($\lambda \neq 0$) such that $[m]\lambda = 0$ in the quotient module. 

This equality $[m\lambda] = 0$ implies that:
\begin{equation*}
    m\lambda \in \Theta(M)
\end{equation*}
Since $m\lambda$ is itself a torsion element of $M$, there must exist some non-zero scalar $\mu \in R$ ($\mu \neq 0$) such that:
\begin{equation*}
    (m\lambda)\mu = 0
\end{equation*}
By the associativity of the module action, we have $m(\lambda\mu) = 0$. Since we are working over an \textbf{integral domain} and both $\lambda \neq 0$ and $\mu \neq 0$, it follows that their product $\lambda\mu \neq 0$.

Since there exists a non-zero scalar $\lambda\mu$ that annihilates $m$, we conclude that $m \in \Theta(M)$. Therefore, in the quotient module:
\begin{equation*}
    [m] = 0
\end{equation*}
Thus, $\Theta(M/\Theta(M))$ consists only of the zero element, which means $M/\Theta(M)$ is torsion-free.
\end{proof}

\hspace{0.5cm}

Okay so that means for any module $M$ over some integral domain $R$ we get the primitive exact sequence: 
\begin{align*}
    0 \rightarrow \Theta(M) \xrightarrow{i} M \xrightarrow{p} M/\Theta(M) \rightarrow 0
\end{align*}



\subsection{Classification Theorem}
There are 3 steps in the classification of such modules: 
\begin{enumerate}
    \item Seperate the module into its free part and torsion submodule
    \item Break down the torsion module into a direct sum of its constituent $p$-torsion modules. 
    \item Classify the individual $p$-torsion modules. 
\end{enumerate}
So really the name of the game is trying to understand the torsion aspects of the module. 
\begin{theorem}
    \label{f.g. free module}
    Let $M$ be a finitely generated torsion free module over a commutative PID. Then $M$ is free.  
\end{theorem}

To prove this we need two lemmas: 

\begin{lemma}
    Let $R$ be a commutative integral domain and $M$ be a torsion-free f.g. module over $R$. Then $M$ is a submodule of some free module (i.e. there exists an injective $R$-module homomorphism $i : M \to R^n$ for some $n$)
\end{lemma}

\begin{lemma}
    Let $R$ be a commutative PID. Let $M \in Mod_R$ be f.g. If $M$ is a submodule of a free module then it is itself free. 
\end{lemma}

\begin{proof}
    Proof of theorem. $M$ is torsion free implies that $M$ is a submodule of a free module by the first lemma. The second lemma then implies that $M$ itself is free. 
\end{proof}

\begin{theorem}
    (Step 1 of Classification) Let $M$ be a finitely generated module over a commutative principal ideal domain. Then $M \cong \Theta(M) \oplus F$ where $F$ is free and $\Theta(M)$ is the torsion submodule of $M$. 
\end{theorem}
\begin{proof}
    We have the primitive short exact sequence $ 0 \rightarrow \Theta(M) \rightarrow M \rightarrow M/\Theta(M) \rightarrow 0$. Since $M/\Theta(M)$ is torsion-free and finitely generated as a quotient of a finitely generated module, we have that $M/\Theta(M) \cong F$ where $F$ is free. Thus giving us the SES: 
    \begin{align*}
        0 \rightarrow \Theta(M) \rightarrow M \rightarrow F \rightarrow 0
    \end{align*}
    Since $F$ is free, it is projective, so the sequence splits, giving us: 
    \begin{align*}
        0 \rightarrow \Theta(M) \rightarrow \Theta(M) \oplus F \rightarrow F \rightarrow 0 
    \end{align*}
    Hence $M \cong  \Theta(M) \oplus  F$
\end{proof}

So $M$ always splits into its torsion submodule and a free module. Note that either $F$ or $\Theta(M)$ could be zero. Free modules are easy to work with, so now we have to deal with the torsion submodule. For now, let $R$ be a commutative noetherian integral domain (not necessarily principal).

\begin{definition}
    Let $a \in R$ such that $a \neq 0$ and $M \in Mod_R$, then define $\Theta_a(m) = \{m \in M: ma = 0\}$. 
\end{definition}

\begin{proposition}
    \label{Identification of Torsion Module}
    If $M$ is finitely generated over $R$ then $\exists a \in R$ where $a \neq 0$ such that $\Theta(M) = \Theta_a(M)$
\end{proposition}
\begin{proof}
    The direction $\Theta_a(M) \subseteq \Theta(M)$ is obvious for all $a$. Now let $\{m_1, \dots, m_n\}$ be an f.g. set for $\Theta(M)$ (which exists because $R$ being noetherian implies any submodule of a finitely generated module is finitely generated). For each $i$ choose $a_i \in R$, $a_i \neq 0$ such that $m_ia_i = 0$. Take $a = \prod a_i$. Then for all $i$ we have $m_i \cdot a = 0$. Since we are in an integral domain, $a \neq 0$.\\
    Therefore, if $m \in \Theta(M)$ i.e. $m = \sum_{i = 1}^{n}m_i \lambda_i$ with $\lambda_i \in R$, we have that
    \begin{align*}
        m\cdot a = \sum_{i = 1}^{n} m_i a \lambda_i = 0 
    \end{align*}
    So $m \in \Theta_a(M)$, giving $\Theta(M) \subseteq \Theta_a(M)$
\end{proof}

Okay, so we can identify the torsion submodule with a particular $a$-torsion module. Now lets try and break this down a little more. Just before, recall that if $R$ is a commuative ring then for $b,c \in R$ we have a submodule (ideal) of $R$ defined by 

\begin{align*}
    (b,c) = \{ b\lambda + c \mu : \lambda, \mu \in R\}
\end{align*}

And we say that $(b,c)$ is coprime when $(b,c) = (1) = R$

\begin{lemma}
    If $a = bc$ and $(b,c) = 1$ then $\Theta_a(M) = \Theta_b(M) \dot{+} \Theta_c(M)$ where $\dot{+}$ is the internal direct sum\footnote{for a longer discussion on internal vs external direct sum, see the appendix}
\end{lemma}
\begin{proof}
    First prove that $\Theta_a(M) = \Theta_b(M) + \Theta_c(M)$. Since $b$ and $c$ are coprime, we have that $\exists \lambda, \mu \in R$ such that $\lambda b + \mu c = 1_R$. Hence for $m \in \Theta_a(M)$ we have $\lambda b m + \mu c m = m$. \\
    $\lambda b m \in \Theta_c(M)$ since $\lambda b c m = \lambda a m = 0$ and similarly $\mu c m \in \Theta_b(M)$.\\
    Now to prove directness, let $x \in \Theta_b(M) \cap \Theta_c(M)$. Observe 
    \begin{align*}
        x = (\lambda b + \mu c)x = \lambda b x + \mu b x = 0 + 0 = 0
    \end{align*}
    Hence $\Theta_b(M) \cap \Theta_c(M) = \{0\}$
\end{proof}

Now we can finally get to our second step in the classification theorem. Note that here we now do require $R$ to also be a commuatative PID.

\begin{theorem}
    (Step 2 of classification theorem) Let $R$ be a commutative noetherian PID, and let $M$ be a finitely generated module over $R$, then we have for some $a \in R$, $p_1, \dots p_n$ prime, and $e_i \geq 1$.  
    \begin{align*}
        \Theta(M) = \Theta_a(M) = \Theta_{{p_1}^{e_1}}(M) \oplus \dots \oplus \Theta_{{p_n}^{e_n}}(M)
    \end{align*}
\end{theorem}
\begin{proof}
    From \ref{Identification of Torsion Module} we can get some $a \in R\setminus \{0\}$ such that $\Theta(M) = \Theta_a(M)$. Since $R$ is a PID, it is a UFD, and so therefore $a$ can be written as a product of prime factors i.e. $a = p_1^{e_1} \dots p_n^{e_n}$ where $p_1, \dots, p_n$ are pairwise distinct primes, and $e_1, \dots, e_n \in \mathbb{N}$.\\
    If $n=1$, then we have that $\Theta_a(M) = \Theta_{p_1^{e_1}}$ and we are done. Otherwise let $b =  p_1^{e_1} \dots p_{n-1}^{e_{n-1}}$ and $c = q_n^{e_n}$. Clearly $a = bc$ and $b$ and $c$ are coprime, so applying the above lemma gives $\Theta_a(M) = \Theta_b(M) \oplus \Theta_c(M)$. Then proceed by induction on $b$. 
\end{proof}


Now, our problem is reduced into classifying the indivdual $p$-torsion submodules. Luckily, as ever, we have a theorem for this: 

\begin{theorem}(Step 3 of classification theorem)
    Suppose $M = \Theta_{p^e}(M)$ where $p$ is a prime and $e \geq 1$. Then there exists a collection $e_1, \dots, e_m$ where $e_1 \leq \dots \leq e_m$ such that 
    \begin{align*}
        \Theta_{p^e}(M) \cong \bigoplus^{m}_{i=1} R/p^{e_i}
    \end{align*}
\end{theorem}

The proof of this is horrible and does some double induction that requires Baer's theorem so whatever. But once we've proved it, we get our full classification theorem: 

\begin{theorem}
    (Classification theorem) Let $M$ be a finitely generated module over a commutative PID $R$. Then $M = R^n \oplus (\bigoplus^{m}_{i = 1} R/p_i^{e_i})$ for some $m$. 
\end{theorem}
\begin{proof}
    Apply the steps in sequence. 
\end{proof}

This is awesome! The first immediate result we get from this is about the global dimension of particular rings. First lets do a few examples. 

\begin{proposition}
    Let $R$ be a PID, and $p \in R$ irred, then pd($R/{p^{e}}) = 1$.
\end{proposition}
\begin{proof}
    We have the free res
    \begin{align*}
        0 \rightarrow R \xrightarrow{p^n} R \xrightarrow{\epsilon} R/{p^e} \rightarrow 0 
    \end{align*}
    To compute the cohomology, apply the contravariant functor $Hom_R(-R)$ to give (ignoring the term at the end):
    \begin{align*}
        0 \rightarrow Hom_R(R,R) \xrightarrow{(p^n)^*} Hom_R(R,R) \rightarrow 0
    \end{align*}
    which simplifies to
    \begin{align*}
        0 \rightarrow R \xrightarrow{p^n} R \rightarrow
    \end{align*}
    Now we can calculate the cohomology groups. Recall that $H^k = ker(d^*_{k+1}/im(d^*_k))$ so $H^0(R/{p^e},R) = ker(d^*_1) = \{r \in R: p^e \cdot r = 0\} = \{0\}$\\
    $H^1 = ker(d^*_2)/im(d^*_1) = R/(p^e)$. \\
    Then 0 for all higher k since there are no modules left to compute! Therefore cd($R/{p^{e}}) \geq 1$ but the resolution tells us that $pd(M) \leq 1$, so they're both exactly 1. 
\end{proof}

\begin{proposition}
    If $P$ is projective, then $pd(P) = 0$. 
\end{proposition}
\begin{proof}
    Immediate from the projective resolution $0 \rightarrow P \xrightarrow{id} P \rightarrow 0$
\end{proof}

\begin{corollary}
    Let $R$ be a commutative PID, then $gd(R) \leq 1$ (If we consider only finitely generated modules)
\end{corollary}
\begin{proof}
    Let $M$ be a finitely generated $R$-module. Then $M \cong R^n \oplus (\bigoplus^{m}_{i = 1} R/p_i^{e_i})$. Then for all $k \in \mathbb{N}$ and f.g. $N$:
    \begin{align*}
        H^k(M,N) \cong H^k(R^n,N) \oplus \bigoplus^m_{i=1}H^k(R/{p_i^{e_i}}, N)
    \end{align*}
    Since $R^n$ is free it is projective its $cd(R^n) = 0$. Moreover, since $cd(R/{p_i^{e_i}}) = 1$ The above sum is 0 for all $k \geq 2$, and if $k = 1$ and $N = R$ then $H^1(R/{p_i^{e_i}}, R) = R/{p_i^{e_i}}$. So we have no cohomology above $k \geq 2$ and possibly some cohomology at $k=1$. Hence $gd(R) \leq 1$
\end{proof}

\begin{proposition}
    Let $R$ be a commutative PID. If $gd(R) = 0$ iff $R$ is a field.
\end{proposition}
\begin{proof}
    If $gd(R) = 0$ this means that for all $R$-modules, $M$, $H^1(M,N) = 0$. From the above proof, this can only happen when $M$ is torsion free, so all $R$-modules are torsion free. Hence by \ref{TorsionFreeField} $R$ is a field.  \\
    If $R$ is a field, then every module is torsion free, hence $M$ is a free module by the classification theorem, and thus $H^1(M,N) = 0$. Therefore gd($R$) $= 0$. 
\end{proof}

\chapter{Hilbert's Syzygy Theorem}
This chapter lays out a proof of the awesomely named "Hilbert's Syzygy Theorem". We'll do this using the mapping cone (or mapping cylinder) construction.

\noindent If we have a ring $\Lambda$ and a short exact sequence $0 \rightarrow C \xrightarrow{f} D \xrightarrow{\epsilon} E \rightarrow 0$ be an exact sequence of $\Lambda$-modules. If 
\begin{align*}
    C_* = ( \dots \rightarrow C_n \xrightarrow{\partial^C_n} C_{n-1} \rightarrow \dots \rightarrow C_0 \xrightarrow{\partial^C_0} C \rightarrow 0)
\end{align*}
and 
\begin{align*}
    D_* = ( \dots \rightarrow D_n \xrightarrow{\partial^D_n} D_{n-1} \rightarrow \dots \rightarrow D_0 \xrightarrow{\partial^D_0} D \rightarrow 0)
\end{align*}
we proved in \ref{Lifting of Chain Complexes} that there exists a lifting of $f$ to $f_*$ such that $f_*: C_* \to D_*$ is a chain mapping i.e
\[
\begin{tikzcd}
    \dots \arrow[r] & C_n \arrow[r, "\partial^C_n"] \arrow[d, "f_n"]
    & C_{n-1} \arrow[r] \arrow[d, "f_{n-1}"] 
    & \dots \arrow[r]
    & C_0 \arrow[r, "\partial^C_0"] \arrow[d, "f_0"]
    & C \arrow[r] \arrow[d, "f"]
    & 0\\
    \dots \arrow[r] & D_n \arrow[r, "\partial^D_n"] 
    & D_{n-1} \arrow[r] 
    & \dots \arrow[r]
    & D_0 \arrow[r, "\partial^D_0"] 
    & D \arrow[r]
    & 0
\end{tikzcd} 
\]
commutes.\\

\begin{definition}
    Define $M(f)_n = D_n \oplus C_{n-1}$ for all $n \in \mathbb{N}$ and define R-linear maps $\partial^{M(f)}_n: M(f)_n \to M(f)_{n-1}$ by 
    \begin{align*}
        \partial^{M(f)}_n = 
        \begin{pmatrix}
            \partial^D_{n} & (-1)^{n-1}f_{n-1} \\
            0                & \partial^C_{n-1}
        \end{pmatrix}
    \end{align*}
    This is called the \textbf{mapping cylinder} of $f$ 
\end{definition}

It is clearly well defined and simple matrix multiplication will show that $\partial^M_n \circ \partial^M_{n+1} = 0$ so this gives a chain complex
\begin{align*}
    M(f) = (\dots \xrightarrow{\partial^{M(f)}_{n+1}} M(f)_n \xrightarrow{\partial^{M(f)}_{n}} M(f)_{n-1} \xrightarrow{\partial^{M(f)}_{n-1}} \dots)
\end{align*}

Thereby giving us an exact sequence of chain complexes: 
\[
\begin{tikzcd}
    0  \arrow[r] & D_{n+1} \arrow[r] \arrow[d, "\partial^D_{n+1}"] 
    & D_{n+1} \oplus C_n \arrow[r] \arrow[d, "i"]
    & C_n \arrow[r] \arrow[d, "\partial^C_n"]
    & 0\\
    0  \arrow[r] & D_{n} \arrow[r] \arrow[d, phantom, "\vdots"]
    & D_{n} \oplus C_{n-1} \arrow[r] \arrow[d, phantom, "\vdots"]
    & C_{n-1} \arrow[r] \arrow[d, phantom, "\vdots"]
    & 0\\
    0  \arrow[r] & D_0 \arrow[r]
    & D_0 \oplus 0 \arrow[r] 
    & 0 \arrow[r] 
    & 0
\end{tikzcd}
\]
where 
\begin{align*}
    i = 
        \begin{pmatrix}
            \partial^D_{n+1} & (-1)^{n}f_{n} \\
            0                & \partial^C_{n}
        \end{pmatrix}
\end{align*}

We fix the indexing problem at the end by letting $K_n = C_{n-1}, K_0 = 0$ and $\partial^K_n = \partial^C_{n-1}$ giving us an exact sequence of chain complexes: 
\begin{align*}
    0 \rightarrow D_* \rightarrow M(f)_* \rightarrow K_* \rightarrow 0
\end{align*}
Thus by the algebraic Mayer-Vietoris theorem, chain version we get a long exact sequence in homology: 
\begin{align*}
    \dots \rightarrow H_n(D_*) \rightarrow H_n(M(f)_*) \rightarrow H_n(K_*) \rightarrow H_{n-1}(D_*) \rightarrow \dots \\
    \dots \rightarrow H_1(D_*) \rightarrow H_1(M(f)_*) \rightarrow H_1(K_*) \rightarrow H_0(D) \rightarrow H_0(M(f)) \rightarrow 0
\end{align*}

\begin{proposition}
    Suppose we have an exact sequence $0 \rightarrow C \xrightarrow{f} D \xrightarrow{p} E \rightarrow$, projective resolutions $C_*$ and $D_*$ for $C,D$ respectively and a lifting of $f$ to $f_* : C_* \to D_*$. Then the mapping cylinder $M(f)_*$ is a projective resolution of $E$. 
\end{proposition}
\begin{proof}
    Since $C_*$ and $D_*$ are projective resolutions of $C$ and $D$, there homology is zero everywhere execept at degree 0, to see this observe the truncated projective resolution for $C$
    \begin{align*}
        \dots \rightarrow C_1 \xrightarrow{\partial^C_1} C_0 \xrightarrow{\partial^C_0} C \rightarrow 0
    \end{align*}
    "Chopping off" to get $\dots \rightarrow C_1 \xrightarrow{\partial^C_1} C_0 \xrightarrow{\partial^C_0} 0$ and computing homology, we get $H_0(C) = ker(\partial^C_0)/im(\partial^C_1) = C_0/ker(\epsilon)$ where the last part is given by exactness. $C_0/ker(\epsilon) \cong C$ and similarly $H_0(D) \cong D$\\
    So applying this to the long exact sequence in homology given to us by the MV theorem we get for $n \geq 2$
    \begin{align*}
        0 \rightarrow 0 \rightarrow H_n(M(f)) \rightarrow 0 \rightarrow 0 
    \end{align*} 
    So $H_n(M(f)) = 0$ for $n \geq 2$.\\
    For $n \leq 1$ we have 
    \begin{align*}
        0 \rightarrow H_1(M(f)) \rightarrow H_0(C) \xrightarrow{f} \rightarrow H_0(D) \rightarrow H_0(M(f)) \rightarrow 0
    \end{align*}
    which then applying our isomorphisms from above gives us: 
    \begin{align*}
        0 \rightarrow H_1(M(f)) \rightarrow C \xrightarrow{f} D \rightarrow H_0(M(f)) \rightarrow 0
    \end{align*}
    $f$ is injective from the original exact sequence, hence $ker(f) = 0$ implying that $H_1(M(f)) = 0$. By first isomorphism theorem since the map $D \to H_0(M(f))$ is surjective, and the kernel of this map is equal to $im(f)$ by exactness $D/ker(D \to H_0(M(f))) = D/im(f) \cong H_0(M(f))$ and from the original sequence $D/im(f) = E$ we have that $E \cong H_0(M(f))$. \\
    Since the homology for $M(f)$ is zero for all $n$ aside from $n = 0$ where its isomorphic to $E$, this implies $M(f)_*$ is a resolution for $E$.
\end{proof}
With this construction, we are ready to prove Hilbert's Syzygy Theorem. 
\begin{theorem}

    \begin{align*}
        gd(R[t_1, \dots, t_n]) = gd(R) + n
    \end{align*}
\end{theorem}

\section{Two-Variable Case}
In this section, we develop a technique on how to compute a resolution of $\mathbb{Z}$ over $\mathbb{Z}[x,y]$. Recall that we already have a resolution of $\mathbb{Z}$ over $\mathbb{Z}[x]$ given by: 
\begin{align*}
    0 \rightarrow Z[x] \xrightarrow{\cdot x} \mathbb{Z}[x] \xrightarrow{\epsilon} \mathbb{Z} \rightarrow 0
\end{align*}
Where $\epsilon$ is the augmentation map. From this, we have a resolution for $\mathbb{Z}[y]$ over $\mathbb{Z}[x,y]$ given by: 
\begin{align*}
    0 \rightarrow \mathbb{Z}[x,y] \xrightarrow{\cdot x} \mathbb{Z}[x,y] \xrightarrow{\epsilon \otimes 1} \mathbb{Z}[y] \rightarrow 0 
\end{align*}
Where we are regarding $\mathbb{Z}[x] \otimes_\mathbb{Z} \mathbb{Z}[y] = \mathbb{Z}[x,y]$ and so $(\epsilon \otimes 1)(f(x)\otimes g(y)) = f(0) \cdot g(y)$ which are polynomials in $y$ only. Putting this together, we get the following diagram: 
\[
\begin{tikzcd}
    & 0 \arrow[d] & 0 \arrow[d] &\\
    & C_1 : = \mathbb{Z}[x,y] \arrow[d, "\cdot x"] \arrow[r, "\cdot y"] & \mathbb{Z}[x,y] =: D_1 \arrow[d, "\cdot x"] &\\
    & C_0 := \mathbb{Z}[x,y] \arrow[d, "\epsilon \otimes 1"] \arrow[r, "\cdot y"] & \mathbb{Z}[x,y] = :D_0 \arrow[d, "\epsilon \otimes 1"] &\\
    0 \arrow[r] & \mathbb{Z}[y] \arrow[r, "\cdot y"] & \mathbb{Z}[y] \arrow[r] & \mathbb{Z} \arrow[r] & 0
\end{tikzcd}
\]
Where we have lifted the map $\cdot y : \mathbb{Z}[y] \to \mathbb{Z}[y]$ from the bottom to the give maps from $\mathbb{Z}[x,y] \to \mathbb{Z}[x,y]$ as in \ref{Lifting of Chain Complexes}. In this sense, we can just treat $\cdot y$ as a chain mapping between $C_*$ and $D_*$. \\

Now, take the mapping cone $M(\cdot y)$ of this chain mapping. Then $M_2(\cdot y) = D_2 \oplus C_1 = 0 \oplus C_1 = C_1$, $M_1(\cdot y) = D_1 \oplus C_1$, and $M_0 (\cdot y)= D_0 \oplus C_{-1} = D_0 \oplus 0 = D_0$. Thus giving us the sequence: 
\begin{align*}
    C_1 \xrightarrow{\partial_2^M} D_1 \oplus C_0 \xrightarrow{\partial_1^M} D_0
\end{align*}
Where using the definition of the mapping cone: 
\begin{align*}
    \partial^M_2 = 
    \begin{pmatrix}
        \partial^D_2 & (-1)^{2-1}y\\
        0            & \partial_1^C
    \end{pmatrix} \text{ and }
    \partial^M_1 = 
    \begin{pmatrix}
        \partial^D_1 & (-1)^{1-1}y \\
        0            & \partial^C_0
    \end{pmatrix}
\end{align*}
Since $\partial^D_2 = 0$, $\partial^C_1 = \cdot x$ and $\partial^D_1 = \cdot x$, $\partial^C_0 = 0$, we have: 
\begin{align*}
    \partial^M_2 = 
    \begin{pmatrix}
        0 & -y\\
        0 & x
    \end{pmatrix} = 
    \begin{pmatrix}
        -y\\
        x
    \end{pmatrix} \text{ and }
    \partial^M_1 = 
    \begin{pmatrix}
        x & y \\
        0 & 0
    \end{pmatrix} = 
    \begin{pmatrix}
        x & y
    \end{pmatrix}
\end{align*} 
So if we splice this with the original resolution for $\mathbb{Z}$ we get: 
\begin{align*}
    0 \rightarrow \mathbb{Z}[x,y] \xrightarrow{\left(\begin{smallmatrix} -y \\ x \end{smallmatrix}\right)} \mathbb{Z}[x,y] \oplus \mathbb{Z}[x,y] \xrightarrow{(x,y)} \mathbb{Z}[x,y] \xrightarrow{\eta} \mathbb{Z} \rightarrow 0
\end{align*}
Where $\eta$ is the double augmentation map (i.e. sends $x$ and $y$ both to zero). The sequence is exact in the middle positions from the construction of the mapping cone and the fact that $Im((x,y)) = \ker{\eta}$. The maps on the side can easily be shown to be injective/surjective respectively, so we have a resolution for $\mathbb{Z}$ over $\mathbb{Z}[x,y]$. 

Now observe what happens when we dualise\footnote{Note that dualising in this context is shorthand for taking $Hom_{\mathbb{Z}[x,y]}(- , \mathbb{Z}[x,y])$}. Since we are dealing with matrices/vectors, dualising just involves taking the transpose, so we get:
\begin{align*}
    0 \rightarrow \mathbb{Z}[x,y] \xrightarrow{\left(\begin{smallmatrix} x \\ y \end{smallmatrix}\right)} \mathbb{Z}[x,y] \oplus \mathbb{Z}[x,y] \xrightarrow{(-y,x)} \mathbb{Z}[x,y] \rightarrow 0 
\end{align*}
Remembering that dualising is contravariant. Then If we then do the change of variables $x \mapsto -y$ and $y \mapsto x$ (which doesn't change the kernels or images), we get 
\begin{align*}
    0 \rightarrow \mathbb{Z}[x,y] \xrightarrow{\left(\begin{smallmatrix} -y \\ x \end{smallmatrix}\right)} \mathbb{Z}[x,y] \oplus \mathbb{Z}[x,y] \xrightarrow{(x,y)} \mathbb{Z}[x,y] \rightarrow 0 
\end{align*}
Thus recovering the original resolution!

\section{Three-Variable Case}
We now repeat this process to get a resolution of $\mathbb{Z}$ over $\mathbb{Z}[x,y,z]$. Introduce a new variable $z$ and as before take the obvious resolution over $\mathbb{Z}[z]$: 
\begin{align*}
    0 \rightarrow \mathbb{Z}[z] \xrightarrow{\cdot z} \mathbb{Z}[z] \rightarrow \mathbb{Z} \rightarrow 0 
\end{align*}
Now, if we label $C_2 := \mathbb{Z}[x,y,z] =: D_2$, $C_1 := \mathbb{Z}[x,y,z] \oplus \mathbb{Z}[x,y,z] =: D_1$ and $C_0 := \mathbb{Z}[x,y,z] =: D_0$, as in our previous construction, we get the following diagram: 
\[
\begin{tikzcd}
    & 0 \arrow[d] & 0 \arrow[d] &\\
    & C_2 \arrow[d, "\partial^C_2"] \arrow[r, "\cdot z"] & D_2 \arrow[d, "\partial^D_2"]\\ 
    & C_1  \arrow[d, "\partial^C_1"] \arrow[r, "\cdot z"] & D_1 \arrow[d, "\partial^D_1"] &\\
    & C_0  \arrow[d] \arrow[r, "\cdot z"] & D_0 \arrow[d] &\\
    0 \arrow[r] & \mathbb{Z}[z] \arrow[r, "\cdot z"] & \mathbb{Z}[z] \arrow[r] & \mathbb{Z} \arrow[r] & 0
\end{tikzcd}
\]
Where we have treated have lifted $\times z : \mathbb{Z}[z] \to \mathbb{Z}[z]$ into a chain mapping between $C_*$ and $D_*$. Now take the mapping cone $M(z)$ of this chain mapping to give: 
\begin{align*}
    &M_3(z) = D_3 \oplus C_2 = 0 \oplus C_2 = \mathbb{Z}[x,y,z]\\
    &M_2(z) = D_2 \oplus C_1 = \mathbb{Z}[x,y,z] \oplus \mathbb{Z}[x,y,z]^{(2)} = \mathbb{Z}[x,y,z]^{(3)}\\
    &M_1(z) = D_1 \oplus C_0 = \mathbb{Z}[x,y,z]^{(2)} \oplus \mathbb{Z}[x,y,z] = \mathbb{Z}[x,y,z]^{(3)}\\
    &M_0(z) = D_0 \oplus C_{-1} = D_0 \oplus 0 = \mathbb{Z}[x,y,z]
\end{align*}

and as before we get 
\begin{align*}
    \mathbb{Z}[x,y,z] \xrightarrow{\partial^M_3} \mathbb{Z}[x,y,z]^{(3)} \xrightarrow{\partial^M_2} \mathbb{Z}[x,y,z]^{(3)} \xrightarrow{\partial^M_1} \mathbb{Z}[x,y,z]
\end{align*}

Where the maps are given by: 
\begin{align*}
    &\partial^M_3 = 
    \begin{pmatrix}
        \partial^D_3 & (-1)^{3-1}z\\
        0            & \partial^C_2
    \end{pmatrix}
    = 
    \begin{pmatrix}
        z\\
        -y\\
        x
    \end{pmatrix}\\
    &\partial^M_2 =
    \begin{pmatrix}
        \partial^D_2 & -z \\
        0            & \partial^C_1
    \end{pmatrix}
    = 
    \begin{pmatrix}
        -y & -z & 0 \\
        x  & 0  & -z \\
        0  & x  & y 
    \end{pmatrix}\\
    &\partial^M_1 =
    \begin{pmatrix}
        \partial^D_1 z
    \end{pmatrix}
    =
    \begin{pmatrix}
        x & y & z
    \end{pmatrix}
\end{align*}
Recalling that $\partial^D_3 = 0$, $\partial^C_2 = \partial^D_2 = \begin{pmatrix}
    -y \\
    x
\end{pmatrix}$, and $\partial^D_1 = \partial^C_1 = (x \, y)$. If we splice this with our original resolution, we thus get:
\begin{align*}
        0 \rightarrow \mathbb{Z}[x,y,z] \xrightarrow{\left(\begin{smallmatrix} z \\ -y \\ x \end{smallmatrix}\right)} \mathbb{Z}[x,y,z]^{(3)} \xrightarrow{\left(\begin{smallmatrix}
        -y & -z & 0 \\ x  & 0  & -z \\ 0  & x  & y \end{smallmatrix}\right)} \mathbb{Z}[x,y,z]^{(3)} \xrightarrow{\left(\begin{smallmatrix} x & y & z \end{smallmatrix} \right)} \mathbb{Z}[x,y,z] \rightarrow \mathbb{Z} \rightarrow 0 
\end{align*}
Which is a resolution of $\mathbb{Z}$ over $\mathbb{Z}[x,y,z]$ (exactness is tedious to check but just trust me). As before, if we dualise (take the transpose and reverse the order) and then apply a change of variables $z \mapsto x$, $-y \mapsto y$ and $x \mapsto z$, we recover the original resolution.

\section{Category Theory}
This is a small aside on category theory. 
\begin{definition}
    A category \textbf{C} consists of a class of objects and sets of morphisms between them. For every ordered pair $A, B$ of objects, there is a set $\text{Hom}_C(A,B)$ of morphisms from $A$ to $B$. Moreover, for every ordered triple $A, B, C$ of objects there is a \textit{law of composition of morphisms}, i.e. a map 
    \begin{align*}
        \text{Hom}_C(A,B) \times \text{Hom}_C(B,C) \rightarrow \text{Hom}_C(A,C)
    \end{align*}
    where $(f,g) \mapsto gf$. 
\end{definition}
The objects and morphisms also satisfy the following axioms for objects $A, B, C, D$. 
\begin{enumerate}
    \item if $A \neq B$ or $C \neq D$, then $\text{Hom}_C(A,B)$ and $\text{Hom}_C(C,D)$ are disjoint sets. 
    \item Composition of morphisms is associative 
    \item Each object has an identity morphism, i.e. for every object $A$ there is a morphism $1_A \in \text{Hom}_C(A,A)$ such that $f(1_A) = f$ for every $f \in \text{Hom}_C(A,B)$ and $1_Ag = g$ for every $g \in \text{Hom}_C(B, A)$
\end{enumerate}

\begin{definition}
    Let $\mathbf{C,D}$ be categories. We say that $\mathcal{F}$ is a \textbf{covariant functor} from $\mathbf{C}$ to $\mathbf{D}$ if 
    \begin{enumerate}
        \item For every object $A$ in $\mathbf{C}$, $\mathcal{F}(A)$ is an object in $\mathbf{D}$. 
        \item For every $f \in \text{Hom}_\mathbf{C}(A,B)$ we have $\mathcal{F}(f) \in \text{Hom}_\mathbf{D}(\mathcal{F}A, \mathcal{F}B)$.  
    \end{enumerate}
    such that the following axioms are satisfied. 
    \begin{enumerate}
        \item if $gf$ is a composition of morphisms in $\mathbf{C}$, then $\mathcal{F}(gf) = \mathcal{F}(g)\mathcal{F}(f)$
        \item $\mathcal{F}(1_A) = 1_{\mathcal{F}A}$
    \end{enumerate}
\end{definition} 
    
\begin{definition}
    Let $\mathbf{C,D}$ be categories. We say that $\mathcal{F}$ is a \textbf{contravariant functor} from $\mathbf{C}$ to $\mathbf{D}$ if it satisfies the same definition as a covariant functor, but with the two following differences: 
    \begin{enumerate}
        \item for every $f \in \text{Hom}_\mathbf{C}(A,B)$, $\mathcal{F}(f) \in \text{Hom}_\mathbf{D}(\mathcal{F}B, \mathcal{F}A)$
        \item if $gf$ is a composition of morphisms in $\mathbf{C}$, then $\mathcal{F}(gf) = \mathcal{F}(f)\mathcal{F}(g)$ in $\mathbf{D}$. 
    \end{enumerate}
    i.e. contravariant functors reverse the arrows. 
\end{definition}

\begin{example}
    In this course, we will work primarily with 3 categories. 
    \begin{itemize}
        \item $\textbf{Mod}_R$: The category of right modules over a ring $R$. The objects are the modules and the morphisms are all $R$-module homomorphisms (i.e. $Hom_R(M,N)$ where $M$ and $N$ are $R$-modules)
        \item $\textbf{Chain}(\Lambda)$: The category of chain complexes over a ring $\Lambda$. The objects are the sequences (known as chain complexes)
        \begin{align*}
            \dots \rightarrow C_{n+1} \xrightarrow{\partial_{n+1}} C_n \xrightarrow{\partial_n} C_{n-1} \rightarrow \dots
        \end{align*}
        wherein each $C_i$ is a $\Lambda$-module, $\partial_n : C_n \rightarrow C_{n-1}$ is $\Lambda$-linear, and $\partial_n \circ \partial_{n+1} = 0$.\\
        If $C_*$ and $D_*$ are chain complexes, the morphisms are the set of homomorphisms of complexes. That is, a map $\alpha : C_* \rightarrow D_*$ which is a set of homomorphisms $\alpha_n : C_n \rightarrow D_n$ wherein the following diagram commutes: 
        \[
        \begin{tikzcd}
        \cdots \arrow[r] 
        & C_{n+1} \arrow[r, "d_{n+1}"] \arrow[d, "\alpha_{n+1}"] 
        & C_n \arrow[r, "d_n"] \arrow[d, "\alpha_n"] 
        & C_{n-1} \arrow[r] \arrow[d, "\alpha_{n-1}"] 
        & \cdots \\
        \cdots \arrow[r] 
        & D_{n+1} \arrow[r, "d_{n+1}"] 
        & D_n \arrow[r, "d_n"] 
        & D_{n-1} \arrow[r] 
        & \cdots
        \end{tikzcd}
        \]
        \item \textbf{Cochain}($\Lambda$): The category of cochain complexes over a ring $\Lambda$. Essentially the same as \textbf{chain}($\Lambda$)\footnote{Note it is convention to use subscript with \textbf{chain}($\Lambda$) and superscript with \textbf{cochain}($\Lambda$) } but the maps between the sequences go from $\delta^n : C^{n-1} \rightarrow C^n$ and $\delta^{n+1} \circ \delta^n = 0$. 
        \item \textbf{Ab}: The category of abelian groups. The objects are abelian groups and the morphisms are group homomorphisms
    \end{itemize}
\end{example}

\begin{remark}
    All of these categories have functors that go between them. 
    \begin{itemize}
        \item $Hom_R(D, -)$ is a covariant functor from $\textbf{Mod}_R$ to $\textbf{Chain}(\Lambda)$ (or $\textbf{Cochain}(\Lambda)$ via index shifting).
        \item Similarly $Hom_R(-, D)$ is a contravariant functor from $\textbf{Mod}_R$ to $\textbf{Chain}(\Lambda)$ (or $\textbf{Cochain}(\Lambda)$ via index shifting).
        \item \textbf{Homology} given by $H_n(C_*) = ker(\partial_n)/Im(\partial_{n+1})$ is a covariant functor from $\textbf{Chain}(\Lambda)$ to \textbf{Ab}.
        \item Similarly \textbf{Cohomology} given by $H^n(C^*) = ker(\delta^{n+1})/Im(\delta_{n})$ is a contravariant functor from $\textbf{Chain}(\Lambda)$ to \textbf{Ab}.
    \end{itemize}
\end{remark}

Functors also have a property called exactness. 

\begin{definition}
    Let $P$ and $Q$ be abelian categories, and $F : P \to Q$ be a covariant functor. Let $0 \rightarrow A \xrightarrow{f} B \xrightarrow{g} C \rightarrow 0$ be a short exact sequence in $P$. Then we say
    \begin{enumerate}
        \item \textbf{left exact} if $0 \rightarrow F(A) \rightarrow F(B) \rightarrow F(C)$ is exact. 
        \item \textbf{right exact} if $F(A) \rightarrow F(B) \rightarrow F(C) \rightarrow 0$ is exact.
        \item \textbf{exact} if $0 \rightarrow F(A) \rightarrow F(B) \rightarrow F(C) \rightarrow 0$ is exact. 
    \end{enumerate}
\end{definition}

\begin{remark}
    This definition applies similarly to contravariant functors. Particularly we say that a contravariant functor $G: F \to P$ is \textbf{exact} if $0 \rightarrow F(C) \rightarrow F(B) \rightarrow F(A) \rightarrow 0$ is exact. 
\end{remark}

\begin{definition}
    Let $\mathcal{C}$ be some category with objects $A,B,C,D,E$ and $f,g$ morphisms of the category wherein $f: A \to B$, and $g:A \to C$. Then we say that $(f,g)$ has a \textbf{pushout} $D$ if there exists morphisms $i : B \to D$ and $j : C \to D$ such that the $A, B, C, D$ square commutes. Moreover, given morphisms $\beta : B \to E$ and $\gamma : C \to E$ such that $\beta \circ f = \gamma \circ g$ there exists a unique $\delta : D \to E$ making the wider diagram commute. 
    \[
    \begin{tikzcd}
    A \arrow[d,"f"'] \arrow[r,"g"] &
    C \arrow[d,"j"'] \arrow[ddr,bend left,"\gamma"] \\
    B \arrow[r,"i"] \arrow[drr,bend right,"\beta"'] &
    D \arrow[dr,dashed,"\delta"] \\
    && E
    \end{tikzcd}
    \]
\end{definition}

\begin{remark}
    The pullback object is defined as the categorical dual of the pushout. I.e. essentially the arrows are reversed. 
\end{remark}

\begin{proposition}
    The category of \textbf{Mod}$_\Lambda$ has pushouts
\end{proposition}
\begin{proof}
    Given such a diagram:
    \[
    \begin{tikzcd}
    A \arrow[d,"f"'] \arrow[r,"g"] & C \\
    B  
    \end{tikzcd}
    \]
    Let $\lim\limits_{\to}(f,g) = B \oplus C/ {\text{Im}(-f \times g)}$. Define $i : B \to \lim\limits_{\to}(f,g)$ and $j : C \to \lim\limits_{\to}(f,g)$ as the respective inclusions into the quotient space, i.e.
    \begin{align*}
        i(b) = \begin{bmatrix} b \\ 0 \end{bmatrix} \text{ and } j(c) = \begin{bmatrix} 0 \\ c \end{bmatrix}
    \end{align*}
    Our claim is that $\lim\limits_{\to}(f,g)$ is a pushout of $f,g$. Suppose we have $\Lambda$-homomorphisms $\beta: B \to E$ and $\gamma: C \to E$ such that $\beta \circ f = \gamma \circ g$. Then there exists a morphism $(\beta, \gamma) : B \oplus C \to E$ given by $(\beta, \gamma) \begin{pmatrix} x \\ y \end{pmatrix} = \beta(x) + \gamma(y)$. Suppose that $\begin{pmatrix} x \\ y \end{pmatrix} \in \text{Im}(f \times g)$. Then by definition $\exists z \in A$ such that $\begin{pmatrix} -f(z) \\ g(z) \end{pmatrix} \in B \oplus C$. Observe, then 
    \begin{align*}
        (\beta, \gamma) \begin{pmatrix} -f(z) \\ g(z) \end{pmatrix} =& -\beta f(z) + \gamma g (z)\\
                                                                    =& \, 0
    \end{align*}
    By commutativity. Hence $(\beta, \gamma)$ restricted to Im($-f \times g$) is zero. Hence ($\beta, \gamma$) induces a unique homomorphism $\delta = [\beta, \gamma] : \lim\limits_{\to}(f,g) \to E$
\end{proof}

\chapter{scrap}
Basically, we want our module $M$ to be free, because then everything is a lot easier. A lot of the time, this cannot happen, so we create a sequence of free modules leading up to $M$. Suppose, from the previous section, we have a $R$-module $M$. Then we can find a surjective linear map $\alpha_0 : F_0 \rightarrow M$ where $F_0$ is free ( This is the 1st approximation to $M$ being free). Now if $\alpha_0$ is also injective, then $M \cong F_0$ and thus $M$ is free. Suppose not, then we have that $Ker(\alpha_0) \neq 0$. Since this is an $R$-module, we can use the proposition again and find surjective $\alpha_1 : F_1 \rightarrow Ker(F_0)$, giving us the sequence: 
\begin{align*}
    F_1 \xrightarrow{\alpha_1} F_0 \xrightarrow{\alpha_0} M \rightarrow 0
\end{align*}
In general, $Ker(\alpha_1)$ isn't injective either, so you repeat the process. Now lets define this all formally. 


\begin{proposition}
    Let $M$ be an $R$-module. Then there exists an exact sequence of $R$-linear mappings: 
    \begin{align*}
        \dots \rightarrow F_{n+1} \xrightarrow{f_{n+1}} F_n \xrightarrow{f_n} F_{n-1} \rightarrow \dots F_2 \xrightarrow{f_2} F_1 \xrightarrow{f_1} F_0 \xrightarrow{f_0} M \rightarrow \dots 
    \end{align*}
    where each $F_i$ is free. 
\end{proposition}

Such a sequence is called a \textbf{free resolution} of $M$. Sometimes it carries on forever and other times it stops after finite steps. 

\begin{example}
    Lets work in the ring $R = \mathbb{Z}$ and take $M = \mathbb{Z}/2$
\end{example}
